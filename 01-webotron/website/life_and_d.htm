א. עולם החיים

 

"המוות" איננו בעולם המתים,

"המוות" הוא בעולם החיים -

כי רק חיים עתידים למות !

מי שלא מת הוא חי,

מי שמת הוא פשוט איננו...

                  

לכן "עולם המתים" לא קיים,

שהרי ההיעדר,

איננו עולם.

כך הוא הסדר:

החי דרכו למות,

המת הוא מת,

ואין בו מהות.

ד. תּוֹחֶלֶת הַחַיִּים וְהַמָּוֶת

 ***************

 

תּוֹחֶלֶת הַמָּוֶת:

 

תּוֹחֶלֶת הַמָּוֶת הוֹלֶכֶת עִם תּוֹחֶלֶת הַחַיִּים ;

כְּכָל שֶׁאֲרֻכָּה יוֹתֵר תּוֹחֶלֶת הַחַיִּים,

כָּךְ אֲרֻכָּה יוֹתֵר תּוֹחֶלֶת הַמָּוֶת,

כְּכָל שֶׁקְּצָרָה יוֹתֵר תּוֹחֶלֶת הַחַיִּים,

כָּךְ קְצָרָה יוֹתֵר תּוֹחֶלֶת הַמָּוֶת.

 

  1.הַמָּוֶת  הָאָרֹךְ הַטִּבְעִי

 

הַמָּוֶת הָאָרֹךְ מֻרְכָּב

מֵאֵרוּעֵי מָוֶת קְטַנִּים,

הַהוֹלְכִים וּמִצְטַבְּרִים

לְאֹרֶךְ שְׁנוֹת הַחַיִּים.

 

בַּד בְּבַד שֶׁהַמָּוֶת

הוֹלֵךְ וּמִתְרַבֶּה.

בַּד בְּבַד הַחַיִּים

הוֹלְכִים וּמִתְמַעֲטִים.

 

כְּמוֹ כֵּלִים שְׁלוּבִים

הֵם הַחַיִּים וְהַמָּוֶת,

כַּאֲשֶׁר הַמָּוֶת עוֹלֶה                          

הַחַיִּים יוֹרְדִים.

                                          

2. הַמָּוֶת הַקָּצָר הַלֹּא טִבְעִי

 

כְּשֶׁהַחַיִּים קְצָרִים 

הַמָּוֶת שֶׁבְּתוֹכָם

גַּם הוּא קָצָר.

כִּמְעַט וְאֵין

אֵרוּעֵי מָוֶת חֶלְקִיִּים,

בְּחַיִּים קְצָרִים.

לֹא כֵּלִים שְׁלוּבִים,

אֶלָּא כֵּלִים שְׁבוּרִים

הֵם חַיִּים צְעִירִים

שֶׁאֵינָם עוֹד.

הַחַיִּים הַקְּצָרִים שֶנִחְיוּ,

אֵינָם מְבִינִים עוֹד

מָוֶת מַהוּ,

וּבְבוֹאוֹ צוֹעֲקִים הֵם

צְעָקָה נוֹרָאָה,

שֶׁמְּהַדְהֶדֶת עוֹד

שִׁבְעִים שָׁנָה אַחֲרֵיהֶם.

 

3. הַמָּוֶת הָאָרֹךְ וְהַלֹּא טִבְעִי

 

הַזְּמַן מִתְאָרֵךְ כְּמוֹ מַחֲלָה כְּרוֹנִית,

לְלֹא קֵץ וּלְלֹא שׁוּם תַּכְלִית.

כְּאִלּוּ שָׁכַח הַזְּמַן אֶת עַצְמוֹ

וְנִכְנַס לְהַמְתָּנָה אֲרֻכָּה בְּלִי לְהַחְלִיט.

 

בַּהַמְתָּנָה הַמְּמֻשֶּׁכֶת וְהַכְרוֹנִית הַזֹּאת,

הַמָּוֶת נִדְחָס וּמְמַלֵּא אֶת כָּל פִּנּוֹת הַזְּמַן,

בְּצוּרָה שֶׁלֹּא מוֹתִירָה שׁוּם מָקוֹם

שֶׁבּוֹ יְכוֹלִים מְעַט חַיִּים לִשְׁהוֹת,

וְלִהְיוֹת שָׁם רֶגַע אֶחָד עִם עַצְמָן.

                       

הַזְּמַן הוּא זְמַן רָקוּב,

חוֹלֶה לֹא-זְמַנִּי,  

וְאֵין אַף רֶגַע אָהוּב

בְּמֶרְחַב הָאֲנִי.

 

וְחֶלְקֵי זְמַן נוֹרָא,

נוֹשְׁרִים מִגּוּף הַחַיִּים,

כְּמוֹ אֹזֶן וּכְמוֹ אַף,

מִגּוּפוֹ שֶׁל מְצֹרָע.

 

הַזְּמַן הִתְבַּלָּה,

הַגּוּף כִּמְעַט וְלֹא נִמְצָא,

מָוֶת מֻחְלָט לֹא בָּא

וְלַחַיִּים אֵין עוֹד מוֹצָא.

 

4. שִׁיר עַל טֶבַע הַמָּוֶת הָאָרֹךְ הַטִּבְעִי

    לְעֻמַּת  טֶבַע הַמָּוֶת הַקָּצָר הַלֹּא טִבְעִי

 

הַמָּוֶת

הַטִּבְעִי הָאָרֹךְ חוֹלֵף

לְאִטּוֹ מֵעוֹנָה לְעוֹנָה בְּדֶרֶךְ-הִלּוּךְ שֶׁל

צַיָּר עִם סַבְלָנוּת הַצּוֹבֵעַ לְאַט-לְאַט תְּמוּנָה חַיָּה

שֶׁמִּשְׁתַּנָּה וּמַחְלִיפָה צְבָעִים מִיָּרֹק עַז לְצָהֹב וּמִצָּהֹב לְחוּם.

הוּא חוֹצֶה לְאִטּוֹ קַיִץ גָּדוּשׁ וּמָלֵא וּלְאַחַר שֶׁכִּמְעַט דִּלֵּג עַל צְמִיחָה וּפְרִיחָה עַזָּה בָּאָבִיב עִקְבוֹתָיו הוֹלְכִים וּמַעֲמִיקִים בַּבְּשֵׁלוּת הָעֲסִיסִית וְהַנִּגֶּרֶת שֶׁל הַפְּרִי עַד לְרִקְבוֹנוֹ והוּא

מַמְשִׁיךְ בִּצְעָדִים מְתוּנִים לְעֵבֶר עוֹנַת

הַסְּתָו וְעוֹנַת הַחֹרֶף. צְעָדִים שֶׁל:

כְּמִישָׁה, הִצְטַמְּקוּת,

הִתְיַבְּשׁוּת,

וּנְשִׁירָה.

 

לְעֻמָּתוֹ:

הַמָּוֶת הַקָּצָר

הוּא מָוֶת מִיָּדִי.

בְּרוּחַ חָזָק קַר

וּפִתְאוֹמִי-

מְגַלֵּחַ מֵהָעֵץ

כָּל נִצָּן וְעָלָה.

בְּמַשַּׁב רוּחַ

חַד-פַּעֲמִי,

בִּשְׁרִיקָה

וּבִצְוָחָה 

שֶׁל מְקוֹנֶנֶת-

מֵעִיף פֶּרַח,

מַפִּיל פְּרִי,

וְהוֹפֵךְ אֶת

הַחַי לְמֵת.

 

ה. מָוֶת שֶׁל כַּמָּה סוּגֵי חַיִּים

  

מָוֶת הַמְּסַיֵּם אֶת הַחַיִּים,                     

מְסַיֵּם לֹא רַק אֶת הַחַיִּים הַפִיזִיִּים,

כִּי אִם אֶת כָּל סוּגֵי הַחַיִּים

הָעוֹשִׂים אֶת הַחַיִּים לְחַיִּים.

 

לְכָל סוּג שֶׁל חַיִּים

יֵשׁ מָוֶת מִשֶּׁלּוֹ,

וּבְכָל סוּג שֶׁל חַיִּים

הַמָּוֶת הוֹלֵךְ וּמִצְטַבֵּר

מֵאֵרוּעֵי מָוֶת חֶלְקִיִּים.

 

מָוֶת שֶׁל כַּמָּה סוּגֵי חַיִּים:

-----------------------

 

עַל כַּמָּה סוּגֵי חַיִּים,

חָלִים אֵרוּעֵי מָוֶת חֶלְקִיִּים,

הַהוֹלְכִים וּמִצְטַבְּרִים,

לְאֹרֶךְ הַחַיִּים:

עַל הַחַיִּים הַפִיזִיִּים,

עַל הַחַיִּים הַכְרוֹנוֹלוֹגִיִּים,

עַל הַחַיִּים הָאֶסְתֶּטִיִּים,

עַל הַחַיִּים הַנַּפְשִׁיִּים,

עַל הַחַיִּים הַדָּתִיִּים (חַיֵּי-אֱמוּנָה)

עַל הַחַיִּים הַחֶבְרָתִיִּים,

עַל הַחַיִּים הַסְּבִיבָתִיִּים,

עַל הַחַיִּים הַהִיסְטוֹרִיִּים.

 

אֵרוּעֵי הַמָּוֶת הַפִיזִי:

 

הַלְבָּנָה, נְשִׁירָה, הַקְמָטָה,                      

הַפְחָתָה, הָאטָה, הַחְלָשָׁה

וִירִידָה -

פְּעָלִים אֵלֶּה הֵם שׁוֹשְׁבִינִים יְדוּעִים,

בַּתַּהֲלוּכָה וּבָאֵרוּעִים הַמִּתְמַשְּׁכִים,                 

שֶׁל הַמָּוֶת הַפִיזִי.

 

אֵרוּעֵי הַמָּוֶת הַכְרוֹנוֹלוֹגִי:

 

תְּלִישַׁת כָּל יוֹם מֵחַיֶּיךָ,

הוּא עוֹד נֵתַח קָטָן שֶׁל זְמַן,

שֶׁהַמָּוֶת מַחְסִיר מֵחַיֶּיךָ.

 

אֵרוּעֵי הַמָּוֶת הָאֶסְתֶּטִי:

 

גּוּף אֱלִיל וְגוּף אֱלִילָה,

הוֹלֵךְ וּמְאַבֵּד אֱלִילוּתוֹ,

פְּנֵי יָהֵל וּפְנֵי הִלָּה,

הוֹלֵךְ וּמְאַבֵּד הִלָּתוֹ.

 

 

אֵרוּעֵי הַמָּוֶת הַנַּפְשִׁי:

 

כָּל הָאָבְדָנִים לְמִינֵיהֶם,

וְהַכִּשְׁלוֹנוֹת,

כָּל הָרְצוֹנוֹת שֶׁלֹּא הִתְמַלְּאוּ,

וְהַתִּקְווֹת שֶׁנָּגוֹזוּ.

 

אֵרוּעֵי הַמָּוֶת הָאֵמוּנִי:

 

הַכְרָזָתוֹ הַקִּיצוֹנִית שֶׁל נִטְּשָׁה:

"אֱלֹהִים מֵת"!

הָיְתָה הַכְרָזָה אִידֵאוֹלוֹגִית,

אֲבָל לְשׂוֹרְדֵי אושוויץ

 "אֱלֹהִים מֵת"!

הָיְתָה זְעָקָה קִיּוּמִית...

                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                             

אֵרוּעֵי הַמָּוֶת הַחֶבְרָתִי:

 

פְּרֵדָה מִקְּרוֹבִים וִידִידִים וּמֵחַיּוֹת מַחְמָד.

    

אֵרוּעֵי הַמָּוֶת הַסְּבִיבָתִי:

 

כְּגוֹן, הַחְלָפַת מְקוֹם מְגוּרֶיךָ,

שִׁנּוּי דִּירָתְךָ, הִתְיַשְּׁנוּת רָהִיטִים,

הִתְכַּלּוּת חֲפָצִים הַחֲבִיבִים עָלֶיךָ.

עֵצִים שֶׁהִתְבַּשַּׂמְתָּ מֵהֶם וְנִכְרְתוּ.

בָּתִּים רְחוֹבוֹת, עָרִים וַאֲרָצוֹת,

שֶׁהָיוּ מְעֻצָּבִים בְּנוֹפֵי חַיֶּיךָ,

שֶׁנֶּעֶלְמוּ מֵחַיֶּיךָ. וְכַדּוֹמֶה.

 

 

אֵרוּעֵי הַמָּוֶת הַהִיסְטוֹרִי:

 

כְּגוֹן, שִׁנּוּיִים תַּרְבּוּתִיִּים שֶׁלֹּא מְקֻבָּלִים עָלֶיךָ,

שִׁנּוּיֵי מִשְׁטָר שֶׁהָיָה טוֹב בְּעֵינֶיךָ,

נְפִילַת מִפְלָגָה אֲהוּדָה עָלֶיךָ.

כִּבּוּשֵׁי אוֹיֵב אֶת אַרְצְךָ,

וִתּוּר עַל חֶלְקֵי אַרְצְךָ. וְכַדּוֹמֶה.

 

 

 

 

"שמעתי שיש אגודת מכחישי מוות - האימורטליים,

אולי כי הם לא אכלו מעץ הדעת ואולי כי הם חושבים/מרגישים 

כאילו הם חיים בגן עדן...

אם כך, באמת לא כדאי להם לשמוע שיש מוות

ושחזון העצמות היבשות של הנביא יחזקאל הוא אלגוריה

וגם אם לא אלגוריה 

הוא עדיין חזון..."

 

       ו. הַמּוּדָעוּת לַמָּוֶת

      -----------------------

 

1. מוּדָעוּת לַמָּוֶת בְּמֶשֶׁךְ חַיִּים אֲרֻכִּים:

 

בְּמֶשֶׁךְ חַיִּים אֲרֻכִּים

אֵרוּעֵי הַמָּוֶת הַחֶלְקִיִּים,

מְסַפְּקִים הִזְדַּמְּנֻיּוֹת רַבּוֹת,

לָחוּשׁ תְּחוּשׁוֹת מָוֶת חֶלְקִיּוֹת,

לְקַבֵּל אוֹתָן, לְהִתְמוֹדֵד אֵיתָן.

בִּהְיוֹתְךָ מוּדָע לַמָּוֶת אַתָּה מֵעִיד,

שֶׁהַמָּוֶת קַיָּם וְהוּא צָפוּי תָּמִיד.

 

2. אֲנִי חַיָּב לָמוּת:

 

זֶה מַתְחִיל עִם מִלָּה

שֶׁנִּזְרֶקֶת בְּהִזְדַּמְּנֻיּוֹת מַמָּשׁ מִקְרִיּוֹת,

אוֹדוֹת חוֹב קָטָן וְלֹא מַשְׁמָעוּתִי.               

אֲנִי לֹא שָׂם לֵב.

זֶה מַמְשִׁיךְ עִם הַתְרָאוֹת

יוֹתֵר מִזְדַּמְּנוֹת,

אוֹדוֹת חוֹב שֶׁגֹּדֶל לְאִטּוֹ,

אֲנִי בְּקֹשִׁי שָׂם לֵב.

זֶה מִתְקַדֵּם עִם הַתְרָאוֹת

יוֹתֵר וְיוֹתֵר תְּכוּפוֹת,

אוֹדוֹת חוֹב הַהוֹלֵךְ וְטוֹפֵחַ.

אֵין לִי בְּרֵרָה,

אֲנִי כְּבָר שָׂם לֵב,

אַךְ מִיָּד אֲנִי שׁוֹכֵחַ.

מִכָּאן וְאֵילָךְ,

מִפַּעַם לְפַעַם,

אֲנִי מְקַבֵּל בִּקּוּר

וְעוֹד בִּקּוּר,

שֶׁל בַּעַל הַחוֹב,

שֶׁל יְדִידָיו,

שֶׁל מַכָּרָיו,

הַמַּזְכִּירִים לִי,

לִפְעָמִים אַף מַזְהִירִים,

לְבַל אֶתְעַלֵּם.

תְּכִיפוּת הַבִּקּוּרִים,

הוֹלֶכֶת וּגְדֵלָה.  

אֲפִלּוּ שֶׁזֶּה מְאֹד מְאֹד לֹא נָעִים לִי,

כְּבָר לֹא אוּכַל יוֹתֵר לְהַפְנוֹת אֶת הַגַּב,

כִּי אִי אֶפְשָׁר לְהַדְחִיק,

וְאִי אֶפְשָׁר לְהַכְחִישׁ.

אֲנִי חַיָּב !

 

3. הוֹדָעָה אַחַת:

 

כָּל הַהִשְׁתַּנֻּיּוֹת, ההתישנויות, הַהִתְבַּלֻּיּוֹת,

הַפְּרֵדוֹת, הָאָבְדָנִים,

הַיָּמִים הַחוֹלְפִים,

וּשְׁאַר סוּגֵי    

הַמָּוֶת -

הַדּוֹפְקִים אֶצְלִי בְּכָל הַדְּלָתוֹת,

הַנּוֹקְשִׁים אַף בְּחַלּוֹנוֹתַי,

הַמְּצַלְצְלִים לְלֹא הַפְסָקָה,

וְהַמַּשְׁאִירִים לִי הוֹדָעוֹת רַבּוֹת.

כֻּלָּם מְנַסִּים בְּכָל דֶּרֶךְ וּבְכָל עֵת,

לְהוֹדִיעַ לִי רַק הוֹדָעָה אַחַת -

תִּהְיֶה מוּדָע לַמָּוֶת.

 

הַחַד פַּעֲמִיּוּת:

 

הַחַד פַּעֲמִיּוּת שֶׁל הַחַיִּים

מְשַׁקֶּפֶת אֶת הֱיוֹתָם -

כְּבָבוּאָה הַמְּשַׁקֶּפֶת אֶת הַדְּמוּת.

כְּמוֹ שֶׁהִשְׁתַּקְּפוּת דְּמוּתֵנוּ בַּמַּרְאָה -

בְּהוֹפִיעֵנוּ מוּלָהּ,

וְהֵעָלְמוּת דְּמוּתֵנוּ מֵהַמַּרְאָה -

בְּהִסְתַּלְּקוּתֵנוּ מִמֶּנָּה,

אוֹמֶרֶת לָנוּ שֶׁאָכֵן זֹאת הִיא בָּבוּאַת דְּמוּתֵנוּ.

כָּךְ הוֹפָעָתָם שֶׁל חַיֵּינוּ וְהִסְתַּלְּקוּתָם מֵהָעוֹלָם,

אוֹמֶרֶת לָנוּ שֶׁאָכֵן אֵלֶּה הֵם בָּבוּאַת חַיֵּינוּ.

 

 

הַמּוּדָעוּת:

 

הַמּוּדָעוּת לַמָּוֶת הִיא הַמּוּדָעוּת לַחַיִּים.

אֵין מוּדָעוּת לַחַיִּים לְלֹא מוּדָעוּת לַמָּוֶת.

הַמָּוֶת הוּא חֵלֶק מֵהַחַיִּים.

אֵיךְ אֶפְשָׁר לִהְיוֹת מוּדָע לְשַׁלֵּם

מִבְּלִי לִהְיוֹת מוּדָע לַחֲלָקָיו ?

           

כְּכָל שֶׁהַדְּבָרִים נְדִירִים יוֹתֵר,

כָּךְ הֵם יְקָרִים יוֹתֵר.

כְּכָל שֶׁהָאֵרוּעִים הֵם חַד פַּעֲמִיִּים יוֹתֵר,

כָּךְ מֻשְׁקָע יוֹתֵר מַאֲמָץ עַל מְנַת לֹא לְהַחְמִיצָם. 

אַךְ כְּכָל שֶׁהַדְּבָרִים מְצוּיִים יוֹתֵר

כָּךְ גְּדֵלָה יוֹתֵר הָאֲדִישׁוּת כְּלַפֵּיהֶם.

וּכְכָל שֶׁהָאֵרוּעִים זְמִינִים יוֹתֵר,

כָּךְ עוֹבְרִים  עֲלֵיהֶם יוֹתֵר מַהֵר לְסֵדֶר הַיּוֹם.

הַחַיִּים הֵם הַדָּבָר הַנָּדִיר בְּיוֹתֵר,               

אֵרוּעֵי הַחַיִּים הֵם הַחַד פַּעֲמִיִּים בְּיוֹתֵר.

אַךְ הַשִּׁגְרָה, הַהִתְעַלְּמוּת וְהַהַכְחָשָׁה,

עוֹשָׂה אוֹתָם לִמְצוּיִים וְלִזְמִינִים כִּבְיָכוֹל.

 

אֵיךְ אֶפְשָׁר לְהַעֲרִיךְ אֶת הָרֶגַע,

אִם בְּאוֹתוֹ הָרֶגַע,

זוֹכְרִים אֶת הָרְגָעִים שֶׁקָּדְמוּ לוֹ

וּמְצַפִּים לָרְגָעִים שֶׁיָּבוֹאוּ אַחֲרָיו ?

 

אֵיךְ אֶפְשָׁר לְהַעֲרִיךְ אֶת הָאֵרוּעַ,

אִם תּוֹפְסִים אוֹתוֹ כְּעוֹד אֵרוּעַ,

מִתּוּךְ סִדְּרָהּ לֹא נִגְמֶרֶת שֶׁל אֵרוּעִים ?

 

אֵיךְ מִתּוּךְ הַשְּׁקִיעָה בְּתַכְתִּיב הַזִּכָּרוֹן

וּבַדֵּעָה הַקְּדוּמָה הָעוֹלָה מֵהֶעָבָר,

וְאֵיךְ מִתּוּךְ טִיסַת הַחֲרָדָה וְהָאַשְׁלָיָה אֶל הֶעָתִיד,

אֶפְשָׁר לִשְׁהוֹת רַק בָּרֶגַע הַזֶּה,

אוֹ לְתַפְקֵד רַק בְּתוֹךְ הָאֵרוּעַ הַזֶּה  ?

 

אֵיךְ מִתּוּךְ הַחַדְגּוֹנִיּוֹת, נוֹסֶכֶת הַשֵּׁנָה מִכָּאן

וּמִתּוּךְ הָאַשְׁלָיוֹת הַמַּרְדִּימוֹת מִכָּאן,

אֶפְשָׁר לְהִתְעוֹרֵר לָאוֹר יְקָרוֹת ?   

 

אֵיךְ אֶפְשָׁר לִרְאוֹת כָּל רֶגַע - נוֹצֵץ כְּיַהֲלוֹם

וְכָל אֵרוּעַ - זֹהַר כִּפְנִינָה ?           

   

        

      ז.  הַקְרָבַת הַחַיִּים

          -------------------

בְּחִירַת הַחַיִּים עַל יְדֵי הַקְרָבַת הַחַיִּים

 

רַק הַמָּוֶת הַסּוֹגֵר עַל הַגּוּף,

מְאַפְשֵׁר לְהַקְרִיב אֶת הַיָּקָר מִכָּל-

הַקְרָבַת גּוּף הַחַיִּים עַצְמָם.

רַק הַמָּוֶת הַסּוֹגֵר עַל הַזְּמַן,

מְאַפְשֵׁר לְהַקְרִיב אֶת הַיָּקָר מִכָּל-

אֶת הַזְּמַן הַמֵּכִיל חַיִּים.

 

מִי שֶׁמַּקְרִיב חַיָּיו-

עַל קִדּוּשׁ הַשֵּׁם,

אוֹמֵר לָנוּ בְּמוֹתוֹ-

שֶׁלֹּא הָיָה שָׁוֶה לוֹ לִחְיוֹת

חַיִּים לְלֹא אֱלֹהִים !

 

מִי שֶׁמַּקְרִיב אֶת חַיָּיו שֶׁלּוֹ-

כְּדֵי לְהַצִּיל חַיֵּי זוּלָתוֹ,

אוֹמֵר לָנוּ בְּמוֹתוֹ -

שֶׁעֵרֶךְ הַחַיִּים עוֹלֶה                       

עַל עֵרֶךְ הַמָּוֶת !

 

מִי שֶׁמַּקְרִיב אֶת חַיָּיו-

לְמַעַן עֵרֶךְ שֶׁהוּא מַאֲמִין בּוֹ,

אוֹמֵר לָנוּ בְּמוֹתוֹ -

שֶׁלֹּא הָיָה שָׁוֶה לוֹ לִחְיוֹת

חַיִּים לְלֹא עֵרֶךְ.

                   

מִי שֶׁמַּקְרִיב אֶת זְמַנּוֹ,

כְּדֵי לְהַגְשִׁים אַהֲבָה,

אוֹמֵר לָנוּ בְּתֹם זְמַנּוֹ,

שֶׁלֹּא הָיָה שָׁוֶה לוֹ לִחְיוֹת

חַיִּים לְלֹא אַהֲבָה.

 

מִי שֶׁמַּקְרִיב אֶת זְמַנּוֹ,

כְּדֵי לְהַגְשִׁים חֲלוֹם,

אוֹמֵר לָנוּ בְּתֹם זְמַנּוֹ,

שֶׁלֹּא הָיָה שָׁוֶה לוֹ לִחְיוֹת

חַיִּים אֲחֵרִים לְלֹא חַיֵּי חֲלוֹמוֹ.

 

 

מָשָׁל רָחֵל וְלֵאָה

 

הַנְּשׂוּאִים לְלֵאָה הַלֹּא-אֲהוּבָה,

נִכְפּוּ עַל יַעֲקֹב,

לִפְנֵי נִשּׂוּאָיו לְרָחֵל הָאֲהוּבָה,

כִּתְנַאי לָרַע וְלַטּוֹב.

 

שֶׁבַע שָׁנִים, יַעֲקֹב עָבַד,

בַּעֲבוּר לֵאָה,

וְעוֹד שֶׁבַע שָׁנִים, יַעֲקֹב עָבַד,

בַּעֲבוּר רָחֵל.

 

לְמַעַן אַהֲבָה גְּדוֹלָה,

צָרִיךְ לְהַקְרִיב.

הַקְרָבָה גְּדוֹלָה -

מְבִיאָה אַהֲבָה גְּדוֹלָה.

 

     -------  

     

עַזָּה כַּמָּוֶת הָאַהֲבָה !

עַזָּה, אַךְ שְׁבִירָה לַמָּוֶת רָחֵל

וְחַיֶּיהָ קְצָרִים.

רַכָּה וּסְתַגְּלָנִית לַחַיִּים לֵאָה

וְחַיֶּיהָ  אֲרֻכִּים.

   ---------

 

כְּכָל שֶׁאֲרֻכִּים יוֹתֵר הַחַיִּים עִם לֵאָה,

הוֹלֵךְ יַעֲקֹב יוֹתֵר לִקְרָאתָהּ וְשָׁב,

וּכְכָל שֶׁהוּא מְקַבֵּל יוֹתֵר אֶת לֵאָה,

חַיָּיו פּוֹרִיִּים יוֹתֵר וְזָרְעוּ רַב.

            *

 

בִּכְדֵי לִיצֹר יְצִירָה גְּדוֹלָה-

צָרִיךְ לְהַקְרִיב,

הַקְרָבָה גְּדוֹלָה -

מְמַמֶּשֶׁת יְצִירָה גְּדוֹלָה .

*

הַחַיִּים בַּזְּמַן:

 

כַּאֲשֶׁר הַחַיִּים "בִּזְמַן לֵאָה"

מְכִילִים רַק אֶת רָחֵל-

הֵם מַפְסִידִים לְבַסּוֹף                       

גַּם אֶת "זְמַן רָחֵל" וְגַם אֶת "זְמַן לֵאָה".

 

אַךְ כַּאֲשֶׁר הַחַיִּים "בִּזְמַן לֵאָה"

מֻקְדָּשִׁים רַק לְלֵאָה-

הֵם מַרְוִיחִים אֶת יַלְדֵי לֵאָה

בְּאֵיכוּת שֶׁל "זְמַן רָחֵל".

      *      

 

כְּדֵי לִמְצֹא מַשְׁמָעוּת-

צָרִיךְ לְקַבֵּל,

קַבָּלָה אֲמִתִּית -

מְגַלָּה מַשְׁמָעוּת אֲמִתִּית.

1. הַחֶשְׁבּוֹנוֹת

 

פְּתִיחַת הַחֶשְׁבּוֹן הַפְּנִימִי 

עַל יְדֵי האפוטרופוסים הָרִאשׁוֹנִים:

                                       

הַחֶשְׁבּוֹן נִפְתַּח בִּתְחִלַּת הַחַיִּים,

עוֹד בְּטֶרֶם הִתְוַדַּעְתִּי לְעַצְמִי.

שְׁלֹשֶׁת האפוטורפסים,

אִמִּי, אָבִי וֶאֱלֹהֵי-הַמְּצִיאוּת שֶׁנּוֹלַדְתִּי לְתוֹכָהּ,

שָׂמוּ לִי בַּחֶשְׁבּוֹן,

אֶת הַנְּתוּנִים הַגֶּנֶטִיִּים וְהַתּוֹרַשְׁתִּיִּים,

כְּמוֹ גַּם אֶת נְתוּנֵי הַסְּבִיבָה שֶׁלְּתוֹכָהּ הִגַּעְתִּי

בְּהָגִיחִי לָעוֹלָם.

מִכָּאן וָהָלְאָה נִכְנְסוּ לַחֶשְׁבּוֹן

הַמְּאֹרָעוֹת הַטּוֹבִים שֶׁקָּרוּ לִי,

וְהַמְּאֹרָעוֹת הָרָעִים שֶׁקָּרוּ לִי.

כַּאֲשֶׁר הַמְּאֹרָעוֹת הָרָעִים עָלוּ

 בְּעָצְמָתָם עַל הַטּוֹבִים

הִתְפַּתְּחוּ הַמִּינוּסִים בְּחֶשְׁבּוֹנִי.

 

חֶרְדַת הַמִּינוּסִים:

כָּךְ הִתְפַּתְּחָה הַחֲרָדָה הַגְּדוֹלָה מִפְּנֵי הַמִּינוּסִים,

הִתְפַּתֵּחַ שִׁגְעוֹן רְדִיפַת הַמִּינוּסִים.

אֵפוֹא שֶׁרַק הָלַכְתִּי וּלְאָן שֶׁהִגַּעְתִּי

הַמִּינוּסִים רָדְפוּ אוֹתִי.

 

הַחֲרָדָה הִיא אֲחוֹת הַכַּעַס,

הוּא הַשְּׁתַלְּטָן הַגָּדוֹל וְהִיא הַשְּׁתַלְּטָנִית הַגְּדוֹלָה.

הִיא הַמַּזְעִיקָה הָאוֹטוֹמָטִית הַמֻּחְלֶטֶת,

הוּא הַמֵּגִיב הָאוֹטוֹמָטִי הַמֻּחְלָט.

כְּמוֹ שֶׁהוּא בְּהִשְׁתַּלְּטוּתוֹ לֹא נוֹתֵן מָקוֹם לִמְאוּמָה חוּץ מִמֶּנּוּ,

הִיא בְּהִשְׁתַּלְּטוּתָהּ לֹא נוֹתֶנֶת מָקוֹם לִמְאוּמָה חוּץ מִמֶּנָּה.

הִיא (הַחֲרָדָה) לֹא נוֹתֶנֶת שְׁהוּת לִתְפֹּס קְצָת מֶרְחָק מֵהַגּוֹרְמִים לְהוֹפָעָתָהּ. כָּךְ שֶׁבְּמֶשֶׁךְ הַשְּׁהוּת תִּהְיֶה אֶפְשָׁרוּת לְהִכָּנֵס, אוּלַי, לְמַחְשָׁבָה אַחֶרֶת, אוּלַי, לְהִסְתַּכְּלוּת אַחֶרֶת.

אַךְ כִּמְעַט וְאֵין שְׁהוּת כָּזֹאת -

הַחֲרָדָה מְצִיפָה !

כִּמְעַט וְאֵין נְתִינַת הִזְדַּמְּנוּת כָּזֹאת -

הַהֲצָפָה מוֹנַעַת !

לָכֵן הַחֲרָדָה שֶׁיּוֹשֶׁבֶת עַל הַכֵּס הָרוֹדָנוּת מֵאֶתְמוֹל, תַּמְשִׁיךְ לָשֶׁבֶת עַל כֵּס הָרוֹדָנוּת

גַּם הַיּוֹם, וּמָה שֶׁהָיָה הוּא שֶׁיִּהְיֶה.

 

הַחֶשְׁבּוֹן הַשְּׁלִילִי:

 

הַחֶשְׁבּוֹן הַשְּׁלִילִי מִתְנַהֵל

תַּחַת הַפַּחַד וְהַחֲרָדָה,

תָּמִיד בְּאוֹתָהּ הַדֶּרֶךְ,

בְּצוּרָה כִּמְעַט אוֹטוֹמָטִית,

לְלֹא שִׁנּוּיִים.

הַחִשּׁוּבִים הַמִּתְבַּצְּעִים בּוֹ,

חוֹזְרִים כָּל הַזְּמַן

אֲחוֹרַנִּית -

לְגִרְעוֹנוֹת הֶעָבָר,

עַד לַגֵּרָעוֹן הַהִיסְטוֹרִי.

 

חֶשְׁבּוֹן זֶה מִתְנַהֵל

בְּצוּרָה עִוֶּרֶת,

כִּמְעַט וְלֹא נִרְשָׁמוֹת בְּתוֹכוֹ הַכְנָסוֹת חֲדָשׁוֹת,

לְעֻמַּת זֹאת, הַהוֹצָאוֹת נִרְשָׁמוֹת בְּתוֹכוֹ

אַחַת לְאַחַת, חָזוֹר וְרָשוֹם, רָשׁוֹם וְחָזוֹר...

כְּשֶׁכָּל הוֹצָאָה מִתְוַסֶּפֶת בְּאֹפֶן מִיָּדִי

לַגֵּרָעוֹן הַהִיסְטוֹרִי.

 

הַחֶשְׁבּוֹן הַזֶּה נוֹגֵד כַּלְכָּלָה וְנוֹגֵד הִגָּיוֹן.

לֹא נִשְׁמַע כָּמוֹהוּ בְּשׁוּם בַּנְק,

בְּשׁוּם מִפְעָל וּבְשׁוּם שׁוּק.

חֶשְׁבּוֹן הַמַּדְגִּישׁ הֶפְסֵדִים,

אַךְ כִּמְעַט מִתְעַלֵּם מִרְוָחִים.

חֶשְׁבּוֹן שֶׁאֵינוֹ מַמְרִיץ הַשְׁקָעוֹת חֲדָשׁוֹת

וְלֹא מְעוֹדֵד מֶחְקָר וּפִתּוּחַ.

 

פַּחַד הַחֶשְׁבּוֹן וְחֶשְׁבּוֹן הַפַּחַד:

 

כָּךְ בְּדִיּוּק מִתְנַהֵל בְּתוֹכְךָ

הַחֶשְׁבּוֹן הַמֵּמִית.

חֶשְׁבּוֹן שֶׁהַטְּרָאוּמָה יוֹלַדְתּוֹ,

הַחֲרָדָה אִמּוֹ

וְהַפַּחַד שְׁמוֹ.

 

חֲרָדָה מִגֵּרָעוֹן,

מִצִּמְצוּם מֻחְלָט,

מִפְּשִׁיטַת רֶגֶל,

מִלֹּא לִהְיוֹת,

מֵהַמָּוֶת.

 

הַפַּחַד לְהִתְחַיֵּב,

לָקַחַת סִכּוּנִים,

לָתֵת סִכּוּי,

לְקַוּוֹת.

 

הַפַּחַד לְהִסְתַּבֵּךְ,

לְהִכָּשֵׁל,

לְהַפְסִיד,

לְהִשָּׁאֵר חַיָּב.

 

וְהַזְּהִירוּת הַמֻּפְרֶזֶת מִפְּנֵי הוֹצָאָה,

מִפְּנֵי נְתִינָה וּמִפְנֵי קַבָּלָה,

מִפְּנֵי הִשְׁתַּתְּפוּת וּמִפְנֵי הֲנָאָה -

זְהִירוּת חוֹלָנִית מֵעֲלוּת-הַחַיִּים.

 

זֶהוּ חֶשְׁבּוֹן כּוֹלֵא חַיִּים,

מְסַכֵּל אֶפְשָׁרֻיּוֹת,

מֵמִית תִּקְוָה,

מֵמִית אַהֲבָה,

וּמְחַרְחֵר רִיב.

רִיב עִם עַצְמְךָ,

רִיב עִם זוּלָתְךָ,

רִיב עִם הַחַיִּים !

                      

וְהַחֶשְׁבּוֹן הַזֶּה הוּא סָמוּי.

וְהוּא כְּמַחֲלָה לֹא נִרְאֵית,

כְּמוֹ לַחַץ דָּם שָׁקֵט.

גַּם אִם בְּמִקְרֶה הוּא מִתְאַזֵּן -

נִכְנָסִים עֲשָׂרָה גִּרְעוֹנוֹת בְּאוֹתָהּ הָעֵת.

 

הַחֶשְׁבּוֹן הַזֶּה רוֹבֵץ בְּשֶׁקֶט,

טָמוּן עָמֹק כְּמוֹ מוֹקֵשׁ.

נִמְצָא מִחוּץ לִשְׂדֵה הַקֶּשֶׁב,

רָחוֹק מִמּוּדָעוּתוֹ.

הוּא רָחוֹק מִבִּקֹּרֶת פְּעִילָה.

הָעִמּוּת/הָרִיב

כַּאֲשֶׁר הַחֶשְׁבּוֹנוֹת הַסְּמוּיִים שֶׁל הַיְּרִיבִים

עוֹמְדִים בֵּינָם לְבֵין חֶשְׁבּוֹן הָרִיב הַגָּלוּי

וְשְׁנֵי אֲנִיים לֹא טִבְעִיִּים,

אֲנִיים בָּלוֹנִיים,

פַּעַם מְנֻפָּחִים מִדַּי וּפַעַם מְרֻקָּנִים מִדַּי,

עוֹמְדִים: אֲנִי-לֹא-טִבְעִי מוּל אֲנִי-לֹא-טִבְעִי.

אֲנִי-לֹא-שָׁקֵט, מְאֹד מְרֻגָּז,

מוּל אֲנִי-לֹא-שָׁקֵט, מְאֹד מְרֻגָּז.

      *

יֵשׁ צְוָחָה שֶׁל אִיּוּם,

יֶשְׁנָהּ  הִתְנַפְּלוּת,

יֶשְׁנָהּ  מַכָּה,

יֶשְׁנָהּ  אֲנָחָה,

יֶשְׁנָהּ פְּצִיעָה וְיֵשׁ מוּם.

 

זֶה יָכֹל לְהַתְחִיל מִלֹּא כְּלוּם

וּכְבָר אָדָם-עִם-אָדָם

וְעַם עִם עַם

מִתְעַמֵּת.

 

עַל כֵּן רִיב הוּא

הַבֵּן הַבְּכוֹר שֶׁל

הַמָּוֶת.

    *

 

הַחֶשְׁבּוֹן הַגָּלוּי:

 

צָרִיךְ לְפָחוֹת שְׁנַיִם לָרִיב,

שְׁנֵי אֲנָשִׁים אוֹ שְׁתֵּי מְדִינוֹת,

שֶׁיֵּשׁ לָהֶם חֶשְׁבּוֹן זֶה עִם זֶה,

חֶשְׁבּוֹן הָרִיב.

 

מִתּוּךְ הַחֶשְׁבּוֹן הַגָּלוּי שֶׁל הָרִיב,

מוֹצִיאִים הַיְּרִיבִים,

אֶת רְשִׁימַת טַעֲנוֹתֵיהֶם

וּמְטִיחִים אוֹתָם,                           

אִישׁ כְּנֶגֶד אָחִיו.

 

אֶחָד מַכְאִיב,

הַשֵּׁנִי מוֹצִיא אֶת הָרַע 

מִבִּפְנִים וּמִסָּבִיב.

 

הַחֶשְׁבּוֹן הַסָּמוּי 

אִלְמָלֵי הַחֶשְׁבּוֹן הַסָּמוּי -

הָיְתָה נוֹצֶרֶת תִּקְשֹׁרֶת,

הָיְתָה מַגִּיעָה הֲבָנָה

וְהָעִמּוּת הָיָה מִסְתַּיֵּם בְּהַסְכָּמָה.

 

כִּי בִּיסוֹדוֹ, אָדָם לָאָדָם - בֶּן אֱנוֹשׁ הִנֵּהוּ

וְגִלּוּי הַלֵּב הוֹפֵךְ אֶת

הַזְּאֵב הַפְּנִימִי

לַכִּבְשָׂה .

 

אַךְ כְּשֶׁהַזְּאֵב מִסְתַּתֵּר וְהַלֵּב סָגוּר,

אָדָם לָאָדָם - זְאֵב!

 

הַזְּאֵב 

הַמִסְתַּתֵּר

לֹא מַפְסִיק לְחַרְחֵר...

 

חִשּׁוּבִים בֵּינְאִישִׁיִּים בַּחֶשְׁבּוֹן הַסָּמוּי

 

מָה שֶׁאֲנִי מַכְחִישׁ אֵצֶל עַצְמִי,

(שֶׁאֲנִי לֹּא בְּסֵדֶר!")

("הוּא לֹא בְּסֵדֶר!")

וּמָה שֶׁאֲנִי מַכְחִישׁ אֵצֶל חֲבֵרִי,

("שֶׁהוּא בְּסֵדֶר!")

אֲנִי רוֹאֶה אֵצֶל עַצְמִי, 

("אֲנִי בְּסֵדֶר!").

 

אֲנִי מַאֲשִׁים אֶת יְרִיבִי -

הוּא מַכְחִישׁ וּמַאֲשִׁים אוֹתִי.

הוּא מַאֲשִׁים אוֹתִי -

אֲנִי מַכְחִישׁ וּמַאֲשִׁים אוֹתוֹ.

אִינְטֶרְמֶצוֹ

(במנגינה עצובה)

אֲנִי מַרְגִּישׁ מְקֻפָּח,

אֲנִי מַרְגִּישׁ נוֹרָא !

         *

וְכָךְ מֵהַאֲשָׁמָה לַהַכְחָשָׁה,

וּמֵהַכְחָשָׁה לַהַאֲשָׁמָה,

הַצִּפִּיָּה מוּפֶרֶת,

וְהַכָּבוֹד נִלְקַח.

הַנְּפָשׁוֹת הַמְּתֻסְכָּלוֹת,

הַכּוֹעֲסוֹת וְהַתּוֹקְפָנִיּוֹת,

מַכְלִילוֹת וּמַשְׁלִיכוֹת אֶחָד עַל הַשֵּׁנִי,

מַצָּבִים מְעֻוָּתִים

וּרְגָשׁוֹת קְרוּעִים.                           

גַּם כְּשֶׁהָאֶחָד אוֹ הַשֵּׁנִי

מִבְּלִי לָדַעַת, אֵיךְ -

שַׁרְשֶׁרֶת הַהַרְגָּשׁוֹת הָרָעִים

הִתְאָרְכָה כָּל כָּךְ.

הַכִּבְיַכוֹל מְנֻמָּק וְהַכִּבְיַכוֹל הֶגְיוֹנִי

הוא חֶשְׁבּוֹן כִּסּוּי לְחֶשְׁבּוֹן סָמוּי מָרִיר,  

מֻגְלָתִי  וְאַרְסִי.

                                                                      

הָרֶגֶשׁ מְעַוֵּת אֶת הַהִגָּיוֹן

עַד כְּדֵי כָּךְ

שֶׁכָּל מִסְפָּר יַחֲסִי

הוֹפֵךְ לִהְיוֹת מִסְפָּר מֻחְלָט,

הוּא- וְאֵין בִּלְתּוֹ, 

אַחֶרֶת- הַכֹּל הָלַךְ!

 

בְּסִכּוּמוֹ הַקָּשֶׁה וְהַמַּר

(שֶׁל אוֹתוֹ הָאָדָם, הַקְּבוּצָה אוֹ הָעָם),

יֵשׁ כְּלַל בַּרְזֶל אֶחָד:

" מחשבוני לא יִמָּחֵק שום מספר לְעוֹלָם,

כִּי בִּמְחִיקָתוֹ אֶמָּחֵק גַּם אֲנִי !"

               *

 

הָאִיּוֹן: "לָמָּה תַּכֶּה רֵעֲךָ?" :

 

אֲנִי שׂוֹנֵא אֶת בַּעַל-רִיבִי,

מוֹחֵק אוֹתוֹ מִתּוֹדַעְתִּי,

מְאַיֵּן אוֹתוֹ בְּאֹפֶן פְּנִימִי,

וְחַס וְחָלִילָה גַּם בְּאֹפֶן חִיצוֹנִי.

 

בְּאַיְּנִי אֶת בַּעַל-רִיבִי,

 אֲנִי מַשְׁמִיד אֶת הַחוֹזֶה הַלֹּא-כָּתוּב שֶׁבֵּינֵינוּ,

הַמְּיַצֵּג הֵן אֶת עֲרָכַי וְהֵן אֶת עֲרָכָיו.

וְכָךְ כְּשֶׁאֲנִי גּוֹרֵעַ חַיִּים

מִבַּעַל-רִיבִי

אֲנִי גּוֹרֵעַ חַיִּים

מֵעַצְמִי.

 

 

הַפָּרָדוֹכְּס

 

 כְּשֶׁהָעִמּוּת הוּא בִּלְתִּי מִתְפַּשֵּׁר

הַמִּתְעַמְּתִים תּוֹפְסִים, מְבִינִים וּמְפָרְשִׁים

זֶה אֶת זֶה 

בְּצוּרָה הֲפוּכָה.

כְּשֶׁהָאֱמֶת שֶׁל זֶה הוּא הַשֶּׁקֶר שֶׁל זֶה,

וְהַשֶּׁקֶר שֶׁל זֶה הוּא הָאֱמֶת שֶׁל זֶה..

 

הַפָּרָדוֹכְּס הַלֹּא לוֹגִי הַזֶּה 

הוּא הַפָּרָדוֹכְּס

הֲכִי אֱנוֹשִׁי

שֶׁבָּעוֹלָם!

 

פָּרָדוֹכְּס שֶׁהַתּוֹצָאָה הַלּוֹגִית

שֶׁלּוֹ הִיא הַמִּלְחָמָה.

כְּשֶׁהַמִּלְחָמָה זֶה הַמַּעֲשֶׂה

הֲכִי לֹא אֱנוֹשִׁי

שֶׁבָּעוֹלָם!

"אוֹי וַאֲבוֹי,

שֶׁהָעוֹלָם מָלֵא

אוֹרוֹת וּמְאוֹרוֹת נִפְלָאִים,

וְהַיָּד הַקְּטַנָּה -

עוֹמֶדֶת בִּפְנֵי הָעֵינַיִם

וּמְעַכֶּבֶת מִלִּרְאוֹת

אֶת הָאוֹרוֹת הַגְּדוֹלִים"

                (הבעש"ט(

 

כַּאֲשֶׁר עוֹלָם שָׁלֵם מַסְתִּיר אֶת הָאוֹר,

אוֹ כַּאֲשֶׁר יָד קְטַנָּה מַסְתִּירָה אֶת הָאוֹר,

הַתּוֹצָאָה הִיא אוֹתָהּ תּוֹצָאָה,

בִּשְׁנֵי הַמִּקְרִים אֵין אוֹר.

 

הַהַסְתָּרָה שֶׁיָּד קְטַנָּה יְכוֹלָה לְהַסְתִּיר,

לִפְעָמִים גַּם עוֹלָם שָׁלֵם

לֹא יָכֹל.

 

הַהַשְׁחָתָה שֶׁיָּד קְטַנָּה יְכוֹלָה לְהַשְׁחִית,

לִפְעָמִים גַּם ג'וּנְגֶּל שָׁלֵם

לֹא יָכֹל.

 

הַשָּׁלוֹם שֶׁיָּד קְטַנָּה מוּשֶׁטֶת יְכוֹלָה לְהָבִיא,

לִפְעָמִים גַּם גּוּף גָּדוֹל,

לֹא יָכֹל.

 

בְּאֵין שָׁלוֹם:

 

בְּאֵין שָׁלוֹם נִקְרָעִים חֶלְקֵי חַיִּים,

מֵעַצְמִי וּמִזּוּלָתִי.

הַקּוֹנְפְלִיקְט מְפוֹרֵר,

אֵין לוֹ תְּנוּעָה אֲחִידָה,

סוּס אֶחָד מוֹשֵׁךְ יָמִינָה,

סוּס שֵׁנִי מוֹשֵׁךְ שְׂמֹאלָה,

וּמִתְמַעֲטִים כָּךְ הַכּוֹחוֹת.

תְּנוּעוֹת לֹא-שְׁקֵטוֹת,

קוֹלוֹת צוֹרְמִים,

וְהַקּוֹל הַלֹּא-נָחוּץ,

עוֹשִׂים הַרְבֵּה מְהוּמָה,

מִבִּפְנִים וּמִבַּחוּץ.

    

שָׁלוֹם:

 

הַשָּׁלוֹם מוֹשֵׁךְ בִּתְנוּעָה אֲחִידָה,

בְּכִוּוּן אֶחָד.

הַקּוֹלוֹת מְתֹאָמִים,

וְכָל קוֹל מְבֹרָךְ,

הַשָּׁלוֹם הוּא תְּהוּדָה נֶהֱדֶרֶת,

כְּשֶׁאַח מְדַבֵּר עִם אָח.

 

שָׁלוֹם הוּא יַחַס שָׁלֵם.

לֹא מְחֻלָּק,

לֹא כָּפוּל,

הוּא יַחַס תָּמִים שֶׁל כָּל הָאִישִׁיּוּת.

 

מוּדָעוֹת לְגַנַּב הַשָּׁלוֹם הַפְּנִימִי:

 

"לֹא תִּגְנֹב" - "לֹא תִּגְנֹב אֶת דַּעַת עַצְמְךָ"

                                   (הרבי מקוצק)

 

לְטַפֵּל בַּחֲרָדוֹת, בִּרְגָשׁוֹת דְּחוּיִים,

דּוֹמֶה לְטִפּוּל בְּמַחֲלָה סְמוּיָה.

קֹדֶם כָּל, צָרִיךְ לְהוֹדוֹת בָּהּ

וְלִהְיוֹת מוּדָעִים לְקִיּוּמָהּ.

 

מוּדָעוּת לְכָךְ שֶׁקַּיָּם חֶשְׁבּוֹן פְּנִימִי סָמוּי

שֶׁתַּהֲלִיכִים פְּרוֹבְּלֶמָטִיִּים בַּיַּלְדוּת,

או/ו מְאֹרָעוֹת טְרָאוּמָטִיִּים,

(מַסְפִּיק שֶׁנֶּחְווּ כְּכָאֵלֶּה)

וְעוֹרְרוּ אֶת הַפַּחַד וְהַחֲרָדָה,

שֶׁהוֹסִיפוּ וְרָשְׁמוּ עוֹד וָעוֹד

הֶפְסֵדִים וְכִשְׁלוֹנוֹת.

 

 .         *

מַצָּב שֶׁל שָׁלוֹם

שָׁלוֹם עִם עַצְמִי,

שָׁלוֹם עִם זוּלָתִי,

שָׁלוֹם עִם הַחַיִּים,

הוּא מַצָּב שֶׁל

מוּדָעוּת - 

 

לְנִהוּל מְאֻזָּן שֶׁל חֶשְׁבּוֹן הַנֶּפֶשׁ.

הַכָּרָה בְּחֶשְׁבּוֹן

הַחוֹבָה וְהַזְּכוּת

הַצֵּל וְהָאוֹר

שֶׁבַּנֶּפֶשׁ

וְנִהוּלָם נִהוּל אַמִּיץ וְיָשָׁר.

 

בְּרִשּׁוּם נָכוֹן,

שֶׁל הַכְנָסוֹת וְהוֹצָאוֹת,

רֶוַח וְהֶפְסֵד.

מָה שֶׁיֵּשׁ וּמָה שֶׁאֵין,

שֶׁאֶפְשָׁר לְשַׁנּוֹת

וְשֶׁאִי אֶפְשָׁר לְשַׁנּוֹת.

וּבְקַבָּלַת הֶפְסֵד וְכִשָּׁלוֹן לְלֹא הַכְלָלָה

וְהַצְלָחָה וְנִצָּחוֹן בָּעֲנָוָה

 

                *

אַל תִּסְגֹּר חֶשְׁבּוֹן בְּמִינוּס,

אַל תִּסְגֹּר אֶת עַצְמְךָ,

אַל תִּתְכַּוֵּץ וְאַל תָּנוּס -

שִׂים לֵב לָאֶפְשָׁרֻיּוֹת בְּכוֹחַ

שֶׁאֶפְשָׁר לִפְתֹּחַ.

 

וְאִם חֶשְׁבּוֹנְךָ עוֹמֵד עַל פְּלוּס

וְאַתָּה סוֹפֵר הַצְלָחוֹת,

אַל תָּעוּף עַל עַצְמְךָ.

לֹא עַל הַכֹּל אַתָּה יָכוֹל - -

תַּן תּוֹדָה לַכֹּל-יָכוֹל!

            *

מַצָּב שֶׁל שָׁלוֹם

שָׁלוֹם עִם עַצְמִי,

שָׁלוֹם עִם זוּלָתִי,

שָׁלוֹם עִם הַחַיִּים,

הוּא מַצָּב שֶׁל מוּדָעוּת -

 

לְמַנְגְּנוֹן הַהַכְחָשָׁה וְהַהַאֲשָׁמָה.

מוּדָעוּת שֶׁמְּבִיאָה לַעֲזִיבַת חִשּׁוּבִים

בֵּינְאִישִׁיִּים סְּמוּיִים

וְתוֹצְאָתָהּ:

 

שֶׁדָּבָר שְׁלִילִי שֶׁאֲנִי לֹא מַכְחִישׁ

וּמוֹדֶה בְּקִיּוּמוֹ אֶצְלִי -

אֲנִי לֹא מַשְׁלִיךְ אוֹתוֹ עַל הָאַחֵר!  

וְדָבָר חִיּוּבִי שֶׁאֲנִי חוֹוֶה אֶת קִיּוּמוֹ אֶצְלִי -

אֲנִי מוֹדֶה בְּקִיּוּמוֹ אֵצֶל הָאַחֵר.

  

אֲנִי לֹא מַאֲשִׁים אֶת חֲבֵרִי -

וְאֵין לוֹ צֹרֶךְ לְהַכְחִישׁ

וּלְהַאֲשִׁים אוֹתִי.

הוּא לֹא מַאֲשִׁים אוֹתִי -

וְאֵין לִי צֹרֶךְ לְהַכְחִישׁ

וּלְהַאֲשִׁים אוֹתוֹ.

 

אֵין צֹרֶךְ בְּחוֹמוֹת וּבְמִבְצְרֵי-הֲגָנָה

 

שָׁלוֹם בְּתוֹךְ עַצְמֵנוּ

מְצַמְצֵם כִּמְעַט לְאֶפֶס

הַכְחָשׁוֹת, הַאֲשָׁמוֹת וְהַשְׁלָכוֹת

שֶׁהֵן הַחוֹמוֹת וּמִבְצְרֵי-הַהֲגָנָה שֶׁל "הָאֲנִי".

שֶׁכֵּן בְּמַצָּב שֶׁל שָׁלוֹם פְּנִימִי

מִי צָרִיךְ חוֹמוֹת וּמִבְצָרִים?

  

רִיב מֵמִית שָׁלוֹם מֵחַיֶיה:

 

בְּעוֹד שֶׁרִיב הוּא אִיּוּן

שָׁלוֹם הוּא אִשּׁוּר,

בְּעוֹד שֶׁבָּרִיב יֵשׁ אֲוִירָה שׁוֹלֶלֶת חַיִּים

בְּשָׁלוֹם יֵשׁ אֲוִירָה מְחַיֶּבֶת חַיִּים.

רִיב מֵמִית, שָׁלוֹם מֵחַיֶּיה.

 

"שָׁלוֹם שֶׁל אַמִּיצִים":

 

שָׁלוֹם זֶה לִהְיוֹת מוּדָע לַפַּחַד

מִפְּנֵי הַוִּתּוּר - וּלְוַתֵּר.

שָׁלוֹם זֶה לָשִׂים לֵב לְהַשְׁפָּעַת הַזְּהִירוּת

מִפְּנֵי נְתִינָה - וְלָתֵת.

שָׁלוֹם זֶה לִהְיוֹת מוּדָע לַסִּכּוּן

בִּלְקִיחַת סִכּוּן - וְלָקַחַת סִכּוּן.

 

שָׁלוֹם זֶה לִהְיוֹת מוּדָע -

לַקּוֹלוֹת הַמְּפַחֲדִים מֵהִסְתַּבְּכוּת,

הַמְּפַחֲדִים מֵהֶפְסֵד,

הַחֲרֵדִים מִלְּהִבָּלַע

ולֹּא-לִהְיוֹת

     -

וּבוֹחֲרִים לִהְיוֹת

בְּשָׁלוֹם.

 

שָׁלוֹם הוּא וִיתוּר עַל מִלְחָמָה

 

וִתּוּר עַל מִלְחָמָה הוּא וִתּוּר עַל הַשִּׁמּוּשׁ

בְּאֶמְצָעִים קִיצוֹנִיִּים עַל מְנַת לִשְׁמֹר עַל הָאִינְטֶרֶסִים.

 

לְוַתֵּר עַל הַהַכָּאָה לְטוֹבַת הַהַפְצָרָה,

לְוַתֵּר עַל הַהַפְצָרָה לְטוֹבַת הַדִּבּוּר,

לְוַתֵּר עַל הַדִּבּוּר לְטוֹבַת הַהַקְשָׁבָה.

 

"וְנַפְשִׁי כְּעָפָר לַכֹּל תִּהְיֶה"

(מִתְּפִלַּת "נְצֹר לְשׁוֹנִי מֵרַע"

שֶׁבְּסוֹף שְׁמוֹנֶה-עֶשְׂרֵה).

 

לְוַתֵּר עַל הַכַּעַס לְטוֹבַת הַתִּסְכּוּל,

לְהַעֲדִיף אֶת הָעֲמִידָה-עַל-שֶׁלְּךָ (הָאָסֶרְטִיבִית)

עַל מַצַּב הַתִּסְכּוּל,

לְוַתֵּר עַל הָעֲמִידָה-עַל-שֶׁלְּךָ

לְטוֹבַת עֲזִיבַת הַ"שֶׁלְּךָ"

וּלְטוֹבַת הַקַּבָּלָה בִּבְרָכָה. (ז'אן קְלַיְן)

 

בְּמַצָּב שֶׁל שָׁלוֹם פְּנִימִי

הָאֲנִי הוּא יוֹתֵר אֲמִתִּי,

פָּחוֹת מִתְנַפֵּחַ אוֹ פָּחוֹת מִצְטַמְצֵם.

הוּא מְבַטֵּא אֶת מַהוּתוֹ בְּקֶשֶׁב מָלֵא,

בִּלְתִּי סֶלֶקְטִיבִי וּבִלְתִּי בִּיקוֹרְתִי,

לְעַצְמוֹ, לְזוּלָתוֹ

וְלָעוֹלָם.

צְפִיָּה לְלֹא הִזְדַּהוּת

=========================

תּוֹדָה לֶהָגוּת הָרוּחָנִית וְלַפְּסִיכוֹלוֹגְיָה

הַמְּיֻחֶדֶת בְּמִינָהּ שֶׁל הַמִּזְרָח הָרָחוֹק וּלְמַמְשִׁיכֶיהָ הַמַּעֲרָבִיִּים:

גורדייב, אוספנסקי, ניקול, וז'אן קְלַיְן

שֶׁזָּכִיתִי להצצה בכִּתְבֵיהֶם.

* ========================= *

 

"מַחְשָׁבוֹת לֹא מִצְטָרְפוֹת לְמַעֲשִׂים",

(ע"פ הַהֲלָכָה)

 

אִם בְּכָל זֹאת, צָפוֹת רְשִׁימוֹת מִתּוּךְ הֶרְגֵּל

אִם בְּכָל זֹאת, עוֹלִים חֶשְׁבּוֹנוֹת

וְאִם חִשּׁוּבִים מַמְשִׁיכִים לְהֵחָשֵׁב,

לְלֹא הַזְמָנָה וְנֶגֶד רְצוֹנֵנוּ -

אֲנַחְנוּ לֹא חַיָּבִים לְהִזְדַּהוֹת אִתָּם.

הֵם רְשׁוּמִים בְּדַפֵּי הַזִּכָּרוֹן

שֶׁלֹּא הַהַחְלָטָה וְלֹא הַזְּמַן הִצְלִיחוּ לְמַחֲקָם,

וְהֵם עוֹלִים בד"כ בְּאֹפֶן אוֹטוֹמָטִי כְּאָסוֹצְיַאצְיוֹת.

 

הֵם אוֹטוֹמָטִיִּים! אֲנַחְנוּ לֹא אוֹטוֹמָטִיִּים!

מָה שֶׁאֲנַחְנוּ יְכוֹלִים לַעֲשׂוֹת כְּשֶׁהֵם עוֹלִים;

לַשִׂים לֵב אֲלֵיהֶם וְלִצְפּוֹת,

נִצְפָּה בָּהֶם, אַךְ לֹא נִזְדַּהָה עִמָּם,

וְלֹא נִקַּח אַחְרָיוּת עַל הוֹפָעָתָם.

צְפִיָּה וְהַקְשָׁבָה נוטראלית וְלֹא בִּקָּרְתִּית -

תְּפַזֵּר אֶת הַמַּחְשָׁבוֹת הַלָּלוּ,

כְּמוֹ שֶׁהַעֲלָאַת אוֹר תְּפַזֵּר חֹשֶׁךְ.

 

הָאַחְרָיוּת לְחַיִּים שֶׁל שָׁלוֹם:

הָאַחְרָיוּת לְחַיִּים שֶׁל שָׁלוֹם,

אֵינֶנָּה אַחְרָיוּת לְהִתְעַסְּקוּת עִם חֶשְׁבּוֹנוֹת.

הָאַחְרָיוּת הַיְּחִידָה הַקַּיֶּמֶת בְּחַיִּים שֶׁל שָׁלוֹם,

הִנָּהּ הָאַחְרָיוּת לִשְׁמִירַת הַשָּׁלוֹם !

 

בְּדִיּוּק כְּשֵׁם שֶׁהָאַחְרָיוּת לְחַיִּים בְּרִיאִים,

אֵינֶנָּה אַחְרָיוּת לְהִתְעַסְּקוּת עִם מַחֲלוֹת.

הָאַחְרָיוּת הַיְּחִידָה לְחַיִּים בְּרִיאִים,

הִנָּהּ הָאַחְרָיוּת לִשְׁמִירַת הַבְּרִיאוּת !

    

שָׁלוֹם : רְאִיָּה כּוֹלֶלֶת :

 

הִסְתַּכְּלוּת שֶׁל שָׁלוֹם הִיא הִסְתַּכְּלוּת שְׁלֵמָה,

הִסְתַּכְּלוּת מַשְׁוָה.

הָרִיב הוּא הַדְגָּשַׁת הַהֲפָכִים,

הַשָּׁלוֹם הוּא הַדְגָּשַׁת הָאֶמְצַע.

 

כַּאֲשֶׁר מִתְמַקְּדִים עַל צַד אֶחָד מִיַּד מִתְבַּלֵּט נִגּוּדוֹ, כְּמוֹ בְּנַדְנֵדָה כַּאֲשֶׁר מִתְיַשְּׁבִים עַל צַד אֶחָד מִיַּד עוֹלֵה הַצַּד הַשֵּׁנִי. 

לָכֵן חַיִּים בְּשָׁלוֹם, הִנָּם חַיִּים עִם הִסְתַּכְּלוּת רְחָבָה שֶׁמְּאַפְשֶׁרֶת לִרְאוֹת בַּד בְּבַד אֶת שְׁנֵי הַהֲפָכִים.

*

חַיִּים בְּשָׁלוֹם, הִנָּם חַיִּים עִם קַבָּלָה אֲמִתִּית,

קַבָּלַת הַנִּגּוּדִים הַמְּצִיאוּתִיִּים:

 

שֶׁבֵּינִי וּבֵינְךָ: הֲוָיָתִי וַהֲוָיָתְךָ, שֶׁלְּךָ וְשֶׁלִּי,

דַּעְתְּךָ וְדַעְתִּי, אֱמוּנָתְךָ וֶאֱמוּנָתִי,

אֲמִיתְךָ וַאֲמִתִּי, צִדְקָתְךָ

וְצִדְקָתִי.

 

    *  

וְאוּלַי אֶפְשָׁר לְהִתְפַּיֵּס בְּשָׁלוֹם

גַּם עִם הַשְּׁאֵלָה הַצּוֹעֶקֶת...

 

כַּאֲשֶׁר מְקַבְּלִים אֶת הַחַיִּים וּמִתְמַסְּרִים לָהֶם

אֲזַי יְכוֹלִים גַּם לְהִתְעַלּוֹת

מֵעַל צוֹרְכֵי הַ"אֶגוֹ",

וְלֹא לַעֲשׂוֹת עִנְיָן פְּנִימִי כָּל כָּךְ גּוֹרֵף

מִכָּל רֶוַח וּמִכָּל הֶפְסֵד,

מִכָּל הַצְלָחָה וּמִכָּל כִּשָּׁלוֹן...

 

*

וְאולי אֶפְשָׁר לְהִתְמוֹדֵד קְצָת יוֹתֵר טוֹב

עִם הַשְּׁאֵלָה הַנּוֹרָאִית:

"לָמָּה זֶה קָרָה דַּוְקָא לִי ?"

אוֹ: "לָמָּה דַּוְקָא לִי זֶה לֹא קָרָה ?"

 

וְכַאֲשֶׁר מֻצּוּ כָּל הַבְּרֵרוֹת,

לְקַבֵּל בְּדוּמִיָּה -

ֹ "חֹסֶר מַזָּל" או "מַזָּל רַע".

 

*

 

אַךְ גַּם כַּאֲשֶׁר הַגּוֹרָל מְסַכֵּל לָנוּ

אֶת הַהִזְדַּמְּנֻיּוֹת,

אוֹ שֶׁאֲנַחְנוּ מַחְמִיצִים אוֹתָן;

עֲדַיִן נוֹתֶרֶת לָנוּ הָאֶפְשָׁרוּת

לְהַגִּישׁ

אֶת הַהִזְדַּמְּנֻיּוֹת הַמְּסֻכָּלוֹת וְהַמֻּחְמָצוֹת שֶׁלָּנוּ -

כְּקָרְבָּן לֶאֱלֹהִים.

 

(שמעתי ברדיו את מרסל דיבואה אומר את זה, כשנשאל על חייו האישיים, והאמנתי לו)

 

הַמַּתָּנָה שֶׁאֱלֹהִים מַחְזִיר לָנוּ הִיא

 

קַבָּלַת שְׁלֵמוּת הַחַיִּים

וְקַבָּלַת הַחַיִּים -

בְּשָׁלוֹם.

 

שַׁבַּת שָׁלוֹם

 

אִם הַשַּׁבָּת הִיא אִי שֶׁבֵּין הַזְּמַנִּים -

הַשָּׁלוֹם הוּא אִי שֶׁבֵּין הַלְּבָבוֹת.

אִם הַשַּׁבָּת הִיא שְׁהוּת

בִּמְקוֹם שֶׁהוּא כָּל מָקוֹם -

הַשָּׁלוֹם הוּא שְׁהוּת

בַּיַּחַס שֶׁהוּא כָּל יַחַס.

 

אִם הַשַּׁבָּת הִיא הַקְשָׁבָה לְקוֹל הָעוֹלָם -

הַשָּׁלוֹם הוּא הַקְשָׁבָה לְקוֹל הָאָדָם.

אִם הַשַּׁבָּת הִיא קַבָּלַת מְצִיאוּת הַשַּׁבָּת -

הַשָּׁלוֹם הוּא קַבָּלַת מְצִיאוּת הָאַחֵר.

 

כְּשֵׁם שֶׁהַשַּׁבָּת הָאִידֵאָלִית מַשְׁבִּיתָה

כָּל תְּנַאי וְכָל הַצְדָּקָה

לְחַיִּים שֶׁל עֲשִׂיָּה וְהֶשֵּׂגִים -

כָּךְ הַשָּׁלוֹם הָאִידֵאָלִי מַשְׁבִּית

כָּל תְּנַאי וְכָל הַצְדָּקָה

לַיְּחָסִים הַתְּלוּיִים בַּדָּבָר

וּלְשָׁלוֹם תָּלוּי בַּדָּבָר.

 

הַשָּׁלוֹם הוּא אַהֲבָה, הַשַּׁבָּת הִיא עֹנֶג,

שַׁבַּת-שָׁלוֹם הִנָּהּ אָהֲבָה בַּתַּעֲנוּגִים !

           

 

הַחַיִּים כִּבְחִירָה הַמָּוֶת כְּהֶכְרֵחַ:

 

הַבְּחִירָה מַרְחִיבָה אֶת הַחַיִּים,

בְּכָךְ שֶׁהִיא מְמַמֶּשֶׁת אֶת הַחֹפֶשׁ הַפְּנִימִי.

בְּחִירָה לִפְעָמִים הִיא מַעֲשֶׂה לֹא קַל,

אַךְ בְּחִירָה הִיא הַמַּעֲשֶׂה.

הַבְּחִירָה הִיא עֶצֶם עִנְיָן הָאָדָם

הַיְּכֹלֶת לִבְחֹר הִיא מַתָּנָה אֱלֹהִית.

*

שֶׁהֲרֵי כְּכָל שֶׁהַיְּצוּר

נָמוּךְ יוֹתֵר בְּסֻלַּם הַהִתְפַּתְּחוּת,

יְכֹלֶת הַבְּחִירָה שֶׁלּוֹ

קְטַנָּה יוֹתֵר.

הוּא חַי חַיִּים רִאשׁוֹנִיִּים,

 אִינְסְטִינְקְטִיבִיִּים וְתוּ לֹא.

כְּכָל שֶׁהַחַיִּים עֶלְיוֹנִים יוֹתֵר

כָּךְ הֵם נִבְחָרִים יוֹתֵר.

כְּכָל שֶׁאָנוּ מִתְפַּתְּחִים יוֹתֵר,

כָּךְ הֲבָנָתֵנוּ אֶת עִנְיַן הַבְּחִירָה עָמֹק יוֹתֵר.

*

שָׁעָה שֶׁאֲנַחְנוּ מְבִינִים, שֶׁכָּל רֶגַע מֵחַיֵּינוּ

יָכֹל לְהִבָּחֵר עַל יָדֵנוּ

כַּאֲשֶׁר אֲנַחְנוּ מְקַבְּלִים אוֹתוֹ בִּבְרָכָה.

 

כַּאֲשֶׁר אֲנַחְנוּ מְקַבְּלִים בִּבְרָכָה

אֶת כָּל הַחַיִּים הַנִחְיִים בָּרֶגַע הַזֶּה,

אֶת כָּל הַנִּכְנָס לִשְׂדֵה-הַקֶּשֶׁב בְּרֶגַע זֶה.

 

כַּאֲשֶׁר אֲנַחְנוּ בּוֹחֲרִים

לָשִׂים בַּמֶּרְכָּז אֶת שְׂדֵה-הַקֶּשֶׁב

וְלֹא אֶת עַצְמֵנוּ.

כַּאֲשֶׁר אֲנַחְנוּ בּוֹחֲרִים

לְשַׁוּוֹת אֱלֹהִים לְנֶגְדֵּנוּ

וְלֹא אוֹתָנוּ לְנֶגְדֵּנוּ.

הַבְּחִירָה הִיא לוּז הַחַיִּים:

 

אֵין כְּמוֹ חַיִּים נִבְחָרִים.

עִם חַיִּים נִבְחָרִים,

חַיִּים, כְּמוֹ אוֹהֵב עִם אֲהוּבָתוֹ,

בְּכָל הַזְּמַנִּים וּבְכָל הַמַּצָּבִים,

בְּאֹשֶׁר וּבְסֵבֶל, בַּתַּעֲנוּג וּבַכְּאֵב...

 

כִּי מִי שֶׁבּוֹחֵר בַּחַיִּים,

בּוֹחֵר לְקַבֵּל אוֹתָם בִּבְרָכָה,

לְקַבֵּל אוֹתָם בְּאַהֲבָה.

 

אֵין אֶפְשָׁרוּת לִחְיוֹת בַּמַּצָּב לְלֹא בְּחִירָה

הַחַיִּים לֹא מְאַפְשְׁרִים וָקוּם שֶׁל בְּחִירָה.

הַחַיִּים הֵם מְחוֹזוֹת הַבְּחִירָה,

רַק הַמָּוֶת הוּא מְחוֹז הַוָּקוּם

הַאֵין-בְּחִירָה.

 

מִי שֶׁאֵינוֹ בּוֹחֵר בַּחַיִּים בּוֹחֵר

בְּפָּסִיבִיּוּת, בַּאֲדִישׁוּת.

הוּא בְּעֶצֶם חַי-מֵת.

רַק הַמִּתְאַבֵּד יוֹתֵר

חַי-מֵת מִמֶּנּוּ,

מֵאַחַר וְהַמִּתְאַבֵּד בּוֹחֵר

אֶת בְּחִירָתוֹ הַנּוֹרָאָה לָמוּת.

 

אַף פַּעַם לֹא נוּכַל לְהַסְכִּים

לְמַצָּב שֶׁל חֹסֶר בְּחִירָה.

נִלָּחֵם וְאַף נַקְרִיב אֶת חַיֵּינוּ

לְמַעַן מַצָּב שֶׁל בְּחִירָה.

שֶׁהֲרֵי בְּלָאו הֲכִי מַצָּב שֶׁל

חֹסֶר בְּחִירָה, מָשׁוּל לְמָוֶת -

הַמָּוֶת מְסַמֵּל שִׂיא שֶׁל חֹסֶר בְּחִירָה.

 

הַבְּחִירָה לְוַתֵּר עַל הַסֵּבֶל מְשַׁחְרֶרֶת

אֶנֶרְגְּיַת-חַיִּים

לֹא אֶבְחַר בַּסֵּבֶל - אֲוַתֵּר עָלָיו.

לֹא אֶבְחַר בְּכַעַס - אֲוַתֵּר עָלָיו.

וְאֶזָּהֵר מְאֹד שֶׁלֹּא אֲתַמְרֵן עַצְמִי

לְמַצָּב מְתֻסְכָּל שֶׁל דָּפוּק,

שֶׁל מְקֻפָּח, שֶׁל מִסְכֵּן.

אהייה אָסֶרְטִיבִי (דַּעְתָּן).

אֶעֱמֹד עַל שֶׁלִּי.

רַק לֹא אֲתַמְרֵן אֶת עַצְמִי לְמַצָּב שֶׁל מְקֻפָּח.

בְּחִירָה בַּמַּצָּב שֶׁל מְקֻפָּח

זֶהוּ קַו גְּבוּל שֶׁל בְּחִירָה גְּרוּעָה,

שֶׁהַרְבֵּה פְּעָמִים הִיא גַּם לֹא מַסְפִּיק מוּדַעַת.

קַו שֶׁמֵּאַחֲרָיו נֶחְצָה קַו הַתִּסְכּוּל וְהָאַלִּימוּת,

קַו חֹסֶר הָאוֹנִים, קַו הַיֵּאוּשׁ וְהַדִּכָּאוֹן.

 

אַךְ שֶׁלֹּא אֶתְבַּלְבֵּל

בֵּין מַצָּב-לֹא-נִבְחַר

לְבֵין מַצָּב-כָּפוּי.

 

מַצָּב-כָּפוּי,

זֶה מַצָּב שֶׁאֵין בְּיְכוֹלְתִי לִבְחֹר בּוֹ.

מַצָּב-לֹא-נִבְחַר,

זֶה מַצָּב שֶׁיֵּשׁ בְּיְכוֹלְתִי לִבְחֹר בּוֹ

אַךְ אֲנִי לֹא בּוֹחֵר בּוֹ.

 

כִּי אוּלַי בְּאֹפֶן אוֹבְּיֶקְטִיבִי

הַבְּחִירָה הִיא קָשָׁה.

כְּגוֹן הַבְּחִירָה לְהֵאָבֵק, לְהִלָּחֵם, לְהַקְרִיב.

 

אוֹ\ו אוּלַי הִיא קָשָׁה בְּאֹפֶן סוּבְּיֶקְטִיבִי,

כָּךְ שֶׁצָּרִיךְ לְגַיֵּס הַרְבֵּה כּוֹחוֹת נֶפֶשׁ

עַל מְנַת לִבְחֹר בָּהּ.

לִפְעָמִים צָרִיךְ לָצֵאת מִמַּצַּב הַנּוֹחוֹת,

מֵהַהֶרְגֵּל שֶׁנֶּהֱפַךְ לְטֶבַע.

וְכִמְעַט תָּמִיד צָרִיךְ לְהִתְגַּבֵּר

עַל הַחֲרָדָה מִפְּנֵי חֹסֶר וַדָּאוּת.

 

המצב של הדפוק

 

יִתָּכֵן, שֶׁאֲנִי אוֹ אַתָּה, יוֹדְעִים הַיּוֹם,

שֶׁיָּכֹלְנוּ לִבְחֹר לְשַׁנּוֹת דְּבָרִים, מַצָּבִים,

וְלֹא בָּחַרְנוּ לְשַׁנּוֹת,

אוֹ שֶׁלֹּא הָיָה לָנוּ כּוֹחַ לְשַׁנּוֹתָם.

וְהַיּוֹם אֲנִי אוֹ אַתָּה,

מַרְגִּישׁ מְתֻסְכָּל, דָּפוּק, מְקֻפָּח, מִסְכֵּן.

 

הָרֶוַח הַמִּשְׁנִי

יֵשׁ לָנוּ רֶוַח מִשְׁנִי

מֵהַשְּׁקִיעָה וּמֵהַהִתְבּוֹסְסוּת

בְּתוֹךְ עִסַּת הָרְגָשׁוֹת הָעֲגוּמִים הַלָּלוּ

הָרֶוַח הַמִּשְׁנִי הִנּוֹ,

בָּאַחְזָקָה שֶׁל עֶמְדַּת "הַמַּאֲשִׁים".

עֶמְדָּה שֶׁנּוֹטֶלֶת מֵעֲצְמֵינוּ מִצַּד אֶחָד,

אֶת הָאַחְרָיוּת לְמַצָּבֵינוּ,

וּמִצַּד שֵׁנִי, אֶת עֹמֶס הָעֲבוֹדָה וְהַהַשְׁקָעָה

הַנִּדְרֶשֶׁת מִצִּדֵּנוּ לְשִׁנּוּי הַמַּצָּב.

בְּהֶתְאֵם לְכָךְ, גַּם נוֹתֶנֶת לָנוּ לֶגִיטִימַצְיָה לְהַרְגִּישׁ

שֶׁהַמְּצִיאוּת, אוֹ שֶׁהַשֵּׁם יִתְבָּרֵךְ, אוֹ הַזּוּלָת חַיָּבִים לָנוּ - -

הַרְגָּשָׁה שֶׁמְּדַרְדֶּרֶת אוֹתָנוּ יוֹתֵר וְיוֹתֵר, כְּכָל שֶׁהַזְּמַן עוֹבֵר,

לִנְסִיגָה יַלְדוּתִית שֶׁל

הֱיוֹת-בְּמַצַּב-תְּלוּת.

 

הַבְּחִירָה לְוַתֵּר ולהשתחרר

 

הֲלֹא יֵשׁ בִּיכָלְתֵּנוּ לִבְחֹר לְוַתֵּר

עַל הַרְגָּשַׁת הַקִּפּוּחַ,

עַל הַכַּעַס,

עַל הַסֵּבֶל הַפְּנִימִי.

 

בְּמַצָּב קִיּוּמִי עָגוּם כְּמוֹ זֶה

הֲלֹא אֵין בְּשׁוּרָה נְכוֹנָה וַאֲמִתִּית יוֹתֵר

מֵאֲשֶׁר הַבְּשׂוֹרָה שֶׁיֵּשׁ בִּיכָלְתֵּנוּ

לִבְחֹר לְהִשְׁתַּחְרֵר

מֵהַמַּצָּב הַקִּיּוּמִי הֶעָגוּם שֶׁשָּׁקַעְנוּ בְּתוֹכוֹ.

 

שֶׁיֵּשׁ בִּיכָלְתֵּנוּ לְגַיֵּס אֶת הָאֹמֶץ לִבְחֹר

בְּוִתּוּר עַל מַחְמֶצֶת הַהַרְגָּשׁוֹת הַשְּׁלִילִיּוֹת

                                           

וְלִדְחוֹת אַחַת וּלְתָמִיד אֶת הַהִתְמַכְּרוּת

לָרֶוַח הַמִּשְׁנִי הֶעָלוּב שֶׁיֵּשׁ לָנוּ מֵהֶם.

וּלְשַׁחְרֵר בְּכָךְ אֶת

אֶנֶרְגְּיַת הַחַיִּים הַכְּלוּאָה.

        --------------

 

הָאֹשֶׁר נִמְצָא בַּפְּרָטִים

 

תִּסְתַּפֵּק גַּם בְּאֶפְשָׁרֻיּוֹת קְטַנּוֹת,

אַל תִּהְיֶה שְׁאַפְתָּן מְתֻסְכָּל, הַשּׁוֹאֵף לְהַלֵּךְ בִּגְדוֹלוֹת, כַּאֲשֶׁר קְטַנּוֹת לֹא נֶחְשָׁבוֹת בְּעֵינָיו.

אַל תִּתְנַפֵּחַ, אַל תִּהְיֶה תַּאַוְתָן,

תִּלְמַד לְהִסְתַּפֵּק בְּמוּעָט.

("אֵיזֶהוּ הֶעָשִׁיר הַשָּׂמֵחַ בְּחֶלְקוֹ"

לָמַדְנוּ בְּבֵית הַסֵּפֶר). 

לְעוֹלָם אַל תְּזַלְזֵל בִּפְרָטִים קְטַנִּים,

שִׂים לֵב אֵלֵיהֶם, תִּהְיֶה עֵרָנִי וְקַשּׁוּב

אַף לַקְּטַנְטַנִּים, לְזַעֲרוּרִיִּים שֶׁבַּדְּבָרִים

בִּסְבִיבָתְךָ.

 

בַּחַיִּים, כְּמוֹ בָּאַהֲבָה, כְּכָל שֶׁאַתָּה אוֹהֵב יוֹתֵר-

אַתָּה מִּתְעַכֵּב יוֹתֵר עַל פְּרָטִים קְטַנִּים

וְשׁוֹהֶה עִמָּם יוֹתֵר זְמַן...

 

אֱהֹב אֶת הַחַיִּים. קַבֵּל אוֹתָם בִּבְרָכָה,

קַבֵּל אוֹתָם בְּאַהֲבָה.

לַטֵּף אוֹתָם בְּעֵינֶיךָ, הָאֲזֵן לָהֶם בְּאָזְנֶיךָ,

חוּשׁ אוֹתָם בְּחוּשֶׁיךָ,

נְשׁוֹם אוֹתָם, שְׁהֵה עִמָּהֶם.

 

 

 

           

הַחַיִּים כִּבְחִירָה הַמָּוֶת כְּהֶכְרֵחַ:

 

הַבְּחִירָה מַרְחִיבָה אֶת הַחַיִּים,

בְּכָךְ שֶׁהִיא מְמַמֶּשֶׁת אֶת הַחֹפֶשׁ הַפְּנִימִי.

בְּחִירָה לִפְעָמִים הִיא מַעֲשֶׂה לֹא קַל,

אַךְ בְּחִירָה הִיא הַמַּעֲשֶׂה.

הַבְּחִירָה הִיא עֶצֶם עִנְיָן הָאָדָם

הַיְּכֹלֶת לִבְחֹר הִיא מַתָּנָה אֱלֹהִית.

*

שֶׁהֲרֵי כְּכָל שֶׁהַיְּצוּר

נָמוּךְ יוֹתֵר בְּסֻלַּם הַהִתְפַּתְּחוּת,

יְכֹלֶת הַבְּחִירָה שֶׁלּוֹ

קְטַנָּה יוֹתֵר.

הוּא חַי חַיִּים רִאשׁוֹנִיִּים,

 אִינְסְטִינְקְטִיבִיִּים וְתוּ לֹא.

כְּכָל שֶׁהַחַיִּים עֶלְיוֹנִים יוֹתֵר

כָּךְ הֵם נִבְחָרִים יוֹתֵר.

כְּכָל שֶׁאָנוּ מִתְפַּתְּחִים יוֹתֵר,

כָּךְ הֲבָנָתֵנוּ אֶת עִנְיַן הַבְּחִירָה עָמֹק יוֹתֵר.

*

שָׁעָה שֶׁאֲנַחְנוּ מְבִינִים, שֶׁכָּל רֶגַע מֵחַיֵּינוּ

יָכֹל לְהִבָּחֵר עַל יָדֵנוּ

כַּאֲשֶׁר אֲנַחְנוּ מְקַבְּלִים אוֹתוֹ בִּבְרָכָה.

 

כַּאֲשֶׁר אֲנַחְנוּ מְקַבְּלִים בִּבְרָכָה

אֶת כָּל הַחַיִּים הַנִחְיִים בָּרֶגַע הַזֶּה,

אֶת כָּל הַנִּכְנָס לִשְׂדֵה-הַקֶּשֶׁב בְּרֶגַע זֶה.

 

כַּאֲשֶׁר אֲנַחְנוּ בּוֹחֲרִים

לָשִׂים בַּמֶּרְכָּז אֶת שְׂדֵה-הַקֶּשֶׁב

וְלֹא אֶת עַצְמֵנוּ.

כַּאֲשֶׁר אֲנַחְנוּ בּוֹחֲרִים

לְשַׁוּוֹת אֱלֹהִים לְנֶגְדֵּנוּ

וְלֹא אוֹתָנוּ לְנֶגְדֵּנוּ.

הַבְּחִירָה הִיא לוּז הַחַיִּים:

 

אֵין כְּמוֹ חַיִּים נִבְחָרִים.

עִם חַיִּים נִבְחָרִים,

חַיִּים, כְּמוֹ אוֹהֵב עִם אֲהוּבָתוֹ,

בְּכָל הַזְּמַנִּים וּבְכָל הַמַּצָּבִים,

בְּאֹשֶׁר וּבְסֵבֶל, בַּתַּעֲנוּג וּבַכְּאֵב...

 

כִּי מִי שֶׁבּוֹחֵר בַּחַיִּים,

בּוֹחֵר לְקַבֵּל אוֹתָם בִּבְרָכָה,

לְקַבֵּל אוֹתָם בְּאַהֲבָה.

 

אֵין אֶפְשָׁרוּת לִחְיוֹת בַּמַּצָּב לְלֹא בְּחִירָה

הַחַיִּים לֹא מְאַפְשְׁרִים וָקוּם שֶׁל בְּחִירָה.

הַחַיִּים הֵם מְחוֹזוֹת הַבְּחִירָה,

רַק הַמָּוֶת הוּא מְחוֹז הַוָּקוּם

הַאֵין-בְּחִירָה.

 

מִי שֶׁאֵינוֹ בּוֹחֵר בַּחַיִּים בּוֹחֵר

בְּפָּסִיבִיּוּת, בַּאֲדִישׁוּת.

הוּא בְּעֶצֶם חַי-מֵת.

רַק הַמִּתְאַבֵּד יוֹתֵר

חַי-מֵת מִמֶּנּוּ,

מֵאַחַר וְהַמִּתְאַבֵּד בּוֹחֵר

אֶת בְּחִירָתוֹ הַנּוֹרָאָה לָמוּת.

 

אַף פַּעַם לֹא נוּכַל לְהַסְכִּים

לְמַצָּב שֶׁל חֹסֶר בְּחִירָה.

נִלָּחֵם וְאַף נַקְרִיב אֶת חַיֵּינוּ

לְמַעַן מַצָּב שֶׁל בְּחִירָה.

שֶׁהֲרֵי בְּלָאו הֲכִי מַצָּב שֶׁל

חֹסֶר בְּחִירָה, מָשׁוּל לְמָוֶת -

הַמָּוֶת מְסַמֵּל שִׂיא שֶׁל חֹסֶר בְּחִירָה.

 

הַבְּחִירָה לְוַתֵּר עַל הַסֵּבֶל מְשַׁחְרֶרֶת

אֶנֶרְגְּיַת-חַיִּים

לֹא אֶבְחַר בַּסֵּבֶל - אֲוַתֵּר עָלָיו.

לֹא אֶבְחַר בְּכַעַס - אֲוַתֵּר עָלָיו.

וְאֶזָּהֵר מְאֹד שֶׁלֹּא אֲתַמְרֵן עַצְמִי

לְמַצָּב מְתֻסְכָּל שֶׁל דָּפוּק,

שֶׁל מְקֻפָּח, שֶׁל מִסְכֵּן.

אהייה אָסֶרְטִיבִי (דַּעְתָּן).

אֶעֱמֹד עַל שֶׁלִּי.

רַק לֹא אֲתַמְרֵן אֶת עַצְמִי לְמַצָּב שֶׁל מְקֻפָּח.

בְּחִירָה בַּמַּצָּב שֶׁל מְקֻפָּח

זֶהוּ קַו גְּבוּל שֶׁל בְּחִירָה גְּרוּעָה,

שֶׁהַרְבֵּה פְּעָמִים הִיא גַּם לֹא מַסְפִּיק מוּדַעַת.

קַו שֶׁמֵּאַחֲרָיו נֶחְצָה קַו הַתִּסְכּוּל וְהָאַלִּימוּת,

קַו חֹסֶר הָאוֹנִים, קַו הַיֵּאוּשׁ וְהַדִּכָּאוֹן.

 

אַךְ שֶׁלֹּא אֶתְבַּלְבֵּל

בֵּין מַצָּב-לֹא-נִבְחַר

לְבֵין מַצָּב-כָּפוּי.

 

מַצָּב-כָּפוּי,

זֶה מַצָּב שֶׁאֵין בְּיְכוֹלְתִי לִבְחֹר בּוֹ.

מַצָּב-לֹא-נִבְחַר,

זֶה מַצָּב שֶׁיֵּשׁ בְּיְכוֹלְתִי לִבְחֹר בּוֹ

אַךְ אֲנִי לֹא בּוֹחֵר בּוֹ.

 

כִּי אוּלַי בְּאֹפֶן אוֹבְּיֶקְטִיבִי

הַבְּחִירָה הִיא קָשָׁה.

כְּגוֹן הַבְּחִירָה לְהֵאָבֵק, לְהִלָּחֵם, לְהַקְרִיב.

 

אוֹ\ו אוּלַי הִיא קָשָׁה בְּאֹפֶן סוּבְּיֶקְטִיבִי,

כָּךְ שֶׁצָּרִיךְ לְגַיֵּס הַרְבֵּה כּוֹחוֹת נֶפֶשׁ

עַל מְנַת לִבְחֹר בָּהּ.

לִפְעָמִים צָרִיךְ לָצֵאת מִמַּצַּב הַנּוֹחוֹת,

מֵהַהֶרְגֵּל שֶׁנֶּהֱפַךְ לְטֶבַע.

וְכִמְעַט תָּמִיד צָרִיךְ לְהִתְגַּבֵּר

עַל הַחֲרָדָה מִפְּנֵי חֹסֶר וַדָּאוּת.

 

המצב של הדפוק

 

יִתָּכֵן, שֶׁאֲנִי אוֹ אַתָּה, יוֹדְעִים הַיּוֹם,

שֶׁיָּכֹלְנוּ לִבְחֹר לְשַׁנּוֹת דְּבָרִים, מַצָּבִים,

וְלֹא בָּחַרְנוּ לְשַׁנּוֹת,

אוֹ שֶׁלֹּא הָיָה לָנוּ כּוֹחַ לְשַׁנּוֹתָם.

וְהַיּוֹם אֲנִי אוֹ אַתָּה,

מַרְגִּישׁ מְתֻסְכָּל, דָּפוּק, מְקֻפָּח, מִסְכֵּן.

 

הָרֶוַח הַמִּשְׁנִי

יֵשׁ לָנוּ רֶוַח מִשְׁנִי

מֵהַשְּׁקִיעָה וּמֵהַהִתְבּוֹסְסוּת

בְּתוֹךְ עִסַּת הָרְגָשׁוֹת הָעֲגוּמִים הַלָּלוּ

הָרֶוַח הַמִּשְׁנִי הִנּוֹ,

בָּאַחְזָקָה שֶׁל עֶמְדַּת "הַמַּאֲשִׁים".

עֶמְדָּה שֶׁנּוֹטֶלֶת מֵעֲצְמֵינוּ מִצַּד אֶחָד,

אֶת הָאַחְרָיוּת לְמַצָּבֵינוּ,

וּמִצַּד שֵׁנִי, אֶת עֹמֶס הָעֲבוֹדָה וְהַהַשְׁקָעָה

הַנִּדְרֶשֶׁת מִצִּדֵּנוּ לְשִׁנּוּי הַמַּצָּב.

בְּהֶתְאֵם לְכָךְ, גַּם נוֹתֶנֶת לָנוּ לֶגִיטִימַצְיָה לְהַרְגִּישׁ

שֶׁהַמְּצִיאוּת, אוֹ שֶׁהַשֵּׁם יִתְבָּרֵךְ, אוֹ הַזּוּלָת חַיָּבִים לָנוּ - -

הַרְגָּשָׁה שֶׁמְּדַרְדֶּרֶת אוֹתָנוּ יוֹתֵר וְיוֹתֵר, כְּכָל שֶׁהַזְּמַן עוֹבֵר,

לִנְסִיגָה יַלְדוּתִית שֶׁל

הֱיוֹת-בְּמַצַּב-תְּלוּת.

 

הַבְּחִירָה לְוַתֵּר ולהשתחרר

 

הֲלֹא יֵשׁ בִּיכָלְתֵּנוּ לִבְחֹר לְוַתֵּר

עַל הַרְגָּשַׁת הַקִּפּוּחַ,

עַל הַכַּעַס,

עַל הַסֵּבֶל הַפְּנִימִי.

 

בְּמַצָּב קִיּוּמִי עָגוּם כְּמוֹ זֶה

הֲלֹא אֵין בְּשׁוּרָה נְכוֹנָה וַאֲמִתִּית יוֹתֵר

מֵאֲשֶׁר הַבְּשׂוֹרָה שֶׁיֵּשׁ בִּיכָלְתֵּנוּ

לִבְחֹר לְהִשְׁתַּחְרֵר

מֵהַמַּצָּב הַקִּיּוּמִי הֶעָגוּם שֶׁשָּׁקַעְנוּ בְּתוֹכוֹ.

 

שֶׁיֵּשׁ בִּיכָלְתֵּנוּ לְגַיֵּס אֶת הָאֹמֶץ לִבְחֹר

בְּוִתּוּר עַל מַחְמֶצֶת הַהַרְגָּשׁוֹת הַשְּׁלִילִיּוֹת

                                           

וְלִדְחוֹת אַחַת וּלְתָמִיד אֶת הַהִתְמַכְּרוּת

לָרֶוַח הַמִּשְׁנִי הֶעָלוּב שֶׁיֵּשׁ לָנוּ מֵהֶם.

וּלְשַׁחְרֵר בְּכָךְ אֶת

אֶנֶרְגְּיַת הַחַיִּים הַכְּלוּאָה.

ט.   הַרְחָבַת הַחַיִּים (הַשָּׁלוֹם הַבְּחִירָה)

+++++++++++++++++++++

 

הַיָּד הַקְּטַנָּה:

 

"אוֹי וַאֲבוֹי,

שֶׁהָעוֹלָם מָלֵא

אוֹרוֹת וּמְאוֹרוֹת נִפְלָאִים,

וְהַיָּד הַקְּטַנָּה -

עוֹמֶדֶת בִּפְנֵי הָעֵינַיִם

וּמְעַכֶּבֶת מִלִּרְאוֹת

אֶת הָאוֹרוֹת הַגְּדוֹלִים"

                (הבעש"ט(

 

כַּאֲשֶׁר עוֹלָם שָׁלֵם מַסְתִּיר אֶת הָאוֹר,

אוֹ כַּאֲשֶׁר יָד קְטַנָּה מַסְתִּירָה אֶת הָאוֹר,

הַתּוֹצָאָה הִיא אוֹתָהּ תּוֹצָאָה,

בִּשְׁנֵי הַמִּקְרִים אֵין אוֹר.

 

הַהַסְתָּרָה שֶׁיָּד קְטַנָּה יְכוֹלָה לְהַסְתִּיר,

לִפְעָמִים גַּם עוֹלָם שָׁלֵם

לֹא יָכֹל.

 

הַהַשְׁחָתָה שֶׁיָּד קְטַנָּה יְכוֹלָה לְהַשְׁחִית,

לִפְעָמִים גַּם ג'וּנְגֶּל שָׁלֵם

לֹא יָכֹל.

 

הַשָּׁלוֹם שֶׁיָּד קְטַנָּה מוּשֶׁטֶת יְכוֹלָה לְהָבִיא,

לִפְעָמִים גַּם גּוּף גָּדוֹל,

לֹא יָכֹל.

 

בְּאֵין שָׁלוֹם:

 

בְּאֵין שָׁלוֹם נִקְרָעִים חֶלְקֵי חַיִּים,

מֵעַצְמִי וּמִזּוּלָתִי.

הַקּוֹנְפְלִיקְט מְפוֹרֵר,

אֵין לוֹ תְּנוּעָה אֲחִידָה,

סוּס אֶחָד מוֹשֵׁךְ יָמִינָה,

סוּס שֵׁנִי מוֹשֵׁךְ שְׂמֹאלָה,

וּמִתְמַעֲטִים כָּךְ הַכּוֹחוֹת.

תְּנוּעוֹת לֹא-שְׁקֵטוֹת,

קוֹלוֹת צוֹרְמִים,

וְהַקּוֹל הַלֹּא-נָחוּץ,

עוֹשִׂים הַרְבֵּה מְהוּמָה,

מִבִּפְנִים וּמִבַּחוּץ.

    

שָׁלוֹם:

 

הַשָּׁלוֹם מוֹשֵׁךְ בִּתְנוּעָה אֲחִידָה,

בְּכִוּוּן אֶחָד.

הַקּוֹלוֹת מְתֹאָמִים,

וְכָל קוֹל מְבֹרָךְ,

הַשָּׁלוֹם הוּא תְּהוּדָה נֶהֱדֶרֶת,

כְּשֶׁאַח מְדַבֵּר עִם אָח.

 

שָׁלוֹם הוּא יַחַס שָׁלֵם.

לֹא מְחֻלָּק,

לֹא כָּפוּל,

הוּא יַחַס תָּמִים שֶׁל כָּל הָאִישִׁיּוּת.

 

מוּדָעוֹת לְגַנַּב הַשָּׁלוֹם הַפְּנִימִי:

 

"לֹא תִּגְנֹב" - "לֹא תִּגְנֹב אֶת דַּעַת עַצְמְךָ"

                                   (הרבי מקוצק)

 

לְטַפֵּל בַּחֲרָדוֹת, בִּרְגָשׁוֹת דְּחוּיִים,

דּוֹמֶה לְטִפּוּל בְּמַחֲלָה סְמוּיָה.

קֹדֶם כָּל, צָרִיךְ לְהוֹדוֹת בָּהּ

וְלִהְיוֹת מוּדָעִים לְקִיּוּמָהּ.

 

מוּדָעוּת לְכָךְ שֶׁקַּיָּם חֶשְׁבּוֹן פְּנִימִי סָמוּי

שֶׁתַּהֲלִיכִים פְּרוֹבְּלֶמָטִיִּים בַּיַּלְדוּת,

או/ו מְאֹרָעוֹת טְרָאוּמָטִיִּים,

(מַסְפִּיק שֶׁנֶּחְווּ כְּכָאֵלֶּה)

וְעוֹרְרוּ אֶת הַפַּחַד וְהַחֲרָדָה,

שֶׁהוֹסִיפוּ וְרָשְׁמוּ עוֹד וָעוֹד

הֶפְסֵדִים וְכִשְׁלוֹנוֹת.

 

 .         *

מַצָּב שֶׁל שָׁלוֹם

שָׁלוֹם עִם עַצְמִי,

שָׁלוֹם עִם זוּלָתִי,

שָׁלוֹם עִם הַחַיִּים,

הוּא מַצָּב שֶׁל

מוּדָעוּת - 

 

לְנִהוּל מְאֻזָּן שֶׁל חֶשְׁבּוֹן הַנֶּפֶשׁ.

הַכָּרָה בְּחֶשְׁבּוֹן

הַחוֹבָה וְהַזְּכוּת

הַצֵּל וְהָאוֹר

שֶׁבַּנֶּפֶשׁ

וְנִהוּלָם נִהוּל אַמִּיץ וְיָשָׁר.

 

בְּרִשּׁוּם נָכוֹן,

שֶׁל הַכְנָסוֹת וְהוֹצָאוֹת,

רֶוַח וְהֶפְסֵד.

מָה שֶׁיֵּשׁ וּמָה שֶׁאֵין,

שֶׁאֶפְשָׁר לְשַׁנּוֹת

וְשֶׁאִי אֶפְשָׁר לְשַׁנּוֹת.

וּבְקַבָּלַת הֶפְסֵד וְכִשָּׁלוֹן לְלֹא הַכְלָלָה

וְהַצְלָחָה וְנִצָּחוֹן בָּעֲנָוָה

 

                *

אַל תִּסְגֹּר חֶשְׁבּוֹן בְּמִינוּס,

אַל תִּסְגֹּר אֶת עַצְמְךָ,

אַל תִּתְכַּוֵּץ וְאַל תָּנוּס -

שִׂים לֵב לָאֶפְשָׁרֻיּוֹת בְּכוֹחַ

שֶׁאֶפְשָׁר לִפְתֹּחַ.

 

וְאִם חֶשְׁבּוֹנְךָ עוֹמֵד עַל פְּלוּס

וְאַתָּה סוֹפֵר הַצְלָחוֹת,

אַל תָּעוּף עַל עַצְמְךָ.

לֹא עַל הַכֹּל אַתָּה יָכוֹל - -

תַּן תּוֹדָה לַכֹּל-יָכוֹל!

            *

מַצָּב שֶׁל שָׁלוֹם

שָׁלוֹם עִם עַצְמִי,

שָׁלוֹם עִם זוּלָתִי,

שָׁלוֹם עִם הַחַיִּים,

הוּא מַצָּב שֶׁל מוּדָעוּת -

 

לְמַנְגְּנוֹן הַהַכְחָשָׁה וְהַהַאֲשָׁמָה.

מוּדָעוּת שֶׁמְּבִיאָה לַעֲזִיבַת חִשּׁוּבִים

בֵּינְאִישִׁיִּים סְּמוּיִים

וְתוֹצְאָתָהּ:

 

שֶׁדָּבָר שְׁלִילִי שֶׁאֲנִי לֹא מַכְחִישׁ

וּמוֹדֶה בְּקִיּוּמוֹ אֶצְלִי -

אֲנִי לֹא מַשְׁלִיךְ אוֹתוֹ עַל הָאַחֵר!  

וְדָבָר חִיּוּבִי שֶׁאֲנִי חוֹוֶה אֶת קִיּוּמוֹ אֶצְלִי -

אֲנִי מוֹדֶה בְּקִיּוּמוֹ אֵצֶל הָאַחֵר.

  

אֲנִי לֹא מַאֲשִׁים אֶת חֲבֵרִי -

וְאֵין לוֹ צֹרֶךְ לְהַכְחִישׁ

וּלְהַאֲשִׁים אוֹתִי.

הוּא לֹא מַאֲשִׁים אוֹתִי -

וְאֵין לִי צֹרֶךְ לְהַכְחִישׁ

וּלְהַאֲשִׁים אוֹתוֹ.

 

אֵין צֹרֶךְ בְּחוֹמוֹת וּבְמִבְצְרֵי-הֲגָנָה

 

שָׁלוֹם בְּתוֹךְ עַצְמֵנוּ

מְצַמְצֵם כִּמְעַט לְאֶפֶס

הַכְחָשׁוֹת, הַאֲשָׁמוֹת וְהַשְׁלָכוֹת

שֶׁהֵן הַחוֹמוֹת וּמִבְצְרֵי-הַהֲגָנָה שֶׁל "הָאֲנִי".

שֶׁכֵּן בְּמַצָּב שֶׁל שָׁלוֹם פְּנִימִי

מִי צָרִיךְ חוֹמוֹת וּמִבְצָרִים?

  

רִיב מֵמִית שָׁלוֹם מֵחַיֶיה:

 

בְּעוֹד שֶׁרִיב הוּא אִיּוּן

שָׁלוֹם הוּא אִשּׁוּר,

בְּעוֹד שֶׁבָּרִיב יֵשׁ אֲוִירָה שׁוֹלֶלֶת חַיִּים

בְּשָׁלוֹם יֵשׁ אֲוִירָה מְחַיֶּבֶת חַיִּים.

רִיב מֵמִית, שָׁלוֹם מֵחַיֶּיה.

 

"שָׁלוֹם שֶׁל אַמִּיצִים":

 

שָׁלוֹם זֶה לִהְיוֹת מוּדָע לַפַּחַד

מִפְּנֵי הַוִּתּוּר - וּלְוַתֵּר.

שָׁלוֹם זֶה לָשִׂים לֵב לְהַשְׁפָּעַת הַזְּהִירוּת

מִפְּנֵי נְתִינָה - וְלָתֵת.

שָׁלוֹם זֶה לִהְיוֹת מוּדָע לַסִּכּוּן

בִּלְקִיחַת סִכּוּן - וְלָקַחַת סִכּוּן.

 

שָׁלוֹם זֶה לִהְיוֹת מוּדָע -

לַקּוֹלוֹת הַמְּפַחֲדִים מֵהִסְתַּבְּכוּת,

הַמְּפַחֲדִים מֵהֶפְסֵד,

הַחֲרֵדִים מִלְּהִבָּלַע

ולֹּא-לִהְיוֹת

     -

וּבוֹחֲרִים לִהְיוֹת

בְּשָׁלוֹם.

 

שָׁלוֹם הוּא וִיתוּר עַל מִלְחָמָה

 

וִתּוּר עַל מִלְחָמָה הוּא וִתּוּר עַל הַשִּׁמּוּשׁ

בְּאֶמְצָעִים קִיצוֹנִיִּים עַל מְנַת לִשְׁמֹר עַל הָאִינְטֶרֶסִים.

 

לְוַתֵּר עַל הַהַכָּאָה לְטוֹבַת הַהַפְצָרָה,

לְוַתֵּר עַל הַהַפְצָרָה לְטוֹבַת הַדִּבּוּר,

לְוַתֵּר עַל הַדִּבּוּר לְטוֹבַת הַהַקְשָׁבָה.

 

"וְנַפְשִׁי כְּעָפָר לַכֹּל תִּהְיֶה"

(מִתְּפִלַּת "נְצֹר לְשׁוֹנִי מֵרַע"

שֶׁבְּסוֹף שְׁמוֹנֶה-עֶשְׂרֵה).

 

לְוַתֵּר עַל הַכַּעַס לְטוֹבַת הַתִּסְכּוּל,

לְהַעֲדִיף אֶת הָעֲמִידָה-עַל-שֶׁלְּךָ (הָאָסֶרְטִיבִית)

עַל מַצַּב הַתִּסְכּוּל,

לְוַתֵּר עַל הָעֲמִידָה-עַל-שֶׁלְּךָ

לְטוֹבַת עֲזִיבַת הַ"שֶׁלְּךָ"

וּלְטוֹבַת הַקַּבָּלָה בִּבְרָכָה. (ז'אן קְלַיְן)

 

בְּמַצָּב שֶׁל שָׁלוֹם פְּנִימִי

הָאֲנִי הוּא יוֹתֵר אֲמִתִּי,

פָּחוֹת מִתְנַפֵּחַ אוֹ פָּחוֹת מִצְטַמְצֵם.

הוּא מְבַטֵּא אֶת מַהוּתוֹ בְּקֶשֶׁב מָלֵא,

בִּלְתִּי סֶלֶקְטִיבִי וּבִלְתִּי בִּיקוֹרְתִי,

לְעַצְמוֹ, לְזוּלָתוֹ

וְלָעוֹלָם.

צְפִיָּה לְלֹא הִזְדַּהוּת

=========================

תּוֹדָה לֶהָגוּת הָרוּחָנִית וְלַפְּסִיכוֹלוֹגְיָה

הַמְּיֻחֶדֶת בְּמִינָהּ שֶׁל הַמִּזְרָח הָרָחוֹק וּלְמַמְשִׁיכֶיהָ הַמַּעֲרָבִיִּים:

גורדייב, אוספנסקי, ניקול, וז'אן קְלַיְן

שֶׁזָּכִיתִי להצצה בכִּתְבֵיהֶם.

* ========================= *

 

"מַחְשָׁבוֹת לֹא מִצְטָרְפוֹת לְמַעֲשִׂים",

(ע"פ הַהֲלָכָה)

 

אִם בְּכָל זֹאת, צָפוֹת רְשִׁימוֹת מִתּוּךְ הֶרְגֵּל

אִם בְּכָל זֹאת, עוֹלִים חֶשְׁבּוֹנוֹת

וְאִם חִשּׁוּבִים מַמְשִׁיכִים לְהֵחָשֵׁב,

לְלֹא הַזְמָנָה וְנֶגֶד רְצוֹנֵנוּ -

אֲנַחְנוּ לֹא חַיָּבִים לְהִזְדַּהוֹת אִתָּם.

הֵם רְשׁוּמִים בְּדַפֵּי הַזִּכָּרוֹן

שֶׁלֹּא הַהַחְלָטָה וְלֹא הַזְּמַן הִצְלִיחוּ לְמַחֲקָם,

וְהֵם עוֹלִים בד"כ בְּאֹפֶן אוֹטוֹמָטִי כְּאָסוֹצְיַאצְיוֹת.

 

הֵם אוֹטוֹמָטִיִּים! אֲנַחְנוּ לֹא אוֹטוֹמָטִיִּים!

מָה שֶׁאֲנַחְנוּ יְכוֹלִים לַעֲשׂוֹת כְּשֶׁהֵם עוֹלִים;

לַשִׂים לֵב אֲלֵיהֶם וְלִצְפּוֹת,

נִצְפָּה בָּהֶם, אַךְ לֹא נִזְדַּהָה עִמָּם,

וְלֹא נִקַּח אַחְרָיוּת עַל הוֹפָעָתָם.

צְפִיָּה וְהַקְשָׁבָה נוטראלית וְלֹא בִּקָּרְתִּית -

תְּפַזֵּר אֶת הַמַּחְשָׁבוֹת הַלָּלוּ,

כְּמוֹ שֶׁהַעֲלָאַת אוֹר תְּפַזֵּר חֹשֶׁךְ.

 

הָאַחְרָיוּת לְחַיִּים שֶׁל שָׁלוֹם:

הָאַחְרָיוּת לְחַיִּים שֶׁל שָׁלוֹם,

אֵינֶנָּה אַחְרָיוּת לְהִתְעַסְּקוּת עִם חֶשְׁבּוֹנוֹת.

הָאַחְרָיוּת הַיְּחִידָה הַקַּיֶּמֶת בְּחַיִּים שֶׁל שָׁלוֹם,

הִנָּהּ הָאַחְרָיוּת לִשְׁמִירַת הַשָּׁלוֹם !

 

בְּדִיּוּק כְּשֵׁם שֶׁהָאַחְרָיוּת לְחַיִּים בְּרִיאִים,

אֵינֶנָּה אַחְרָיוּת לְהִתְעַסְּקוּת עִם מַחֲלוֹת.

הָאַחְרָיוּת הַיְּחִידָה לְחַיִּים בְּרִיאִים,

הִנָּהּ הָאַחְרָיוּת לִשְׁמִירַת הַבְּרִיאוּת !

    

שָׁלוֹם : רְאִיָּה כּוֹלֶלֶת :

 

הִסְתַּכְּלוּת שֶׁל שָׁלוֹם הִיא הִסְתַּכְּלוּת שְׁלֵמָה,

הִסְתַּכְּלוּת מַשְׁוָה.

הָרִיב הוּא הַדְגָּשַׁת הַהֲפָכִים,

הַשָּׁלוֹם הוּא הַדְגָּשַׁת הָאֶמְצַע.

 

כַּאֲשֶׁר מִתְמַקְּדִים עַל צַד אֶחָד מִיַּד מִתְבַּלֵּט נִגּוּדוֹ, כְּמוֹ בְּנַדְנֵדָה כַּאֲשֶׁר מִתְיַשְּׁבִים עַל צַד אֶחָד מִיַּד עוֹלֵה הַצַּד הַשֵּׁנִי. 

לָכֵן חַיִּים בְּשָׁלוֹם, הִנָּם חַיִּים עִם הִסְתַּכְּלוּת רְחָבָה שֶׁמְּאַפְשֶׁרֶת לִרְאוֹת בַּד בְּבַד אֶת שְׁנֵי הַהֲפָכִים.

*

חַיִּים בְּשָׁלוֹם, הִנָּם חַיִּים עִם קַבָּלָה אֲמִתִּית,

קַבָּלַת הַנִּגּוּדִים הַמְּצִיאוּתִיִּים:

 

שֶׁבֵּינִי וּבֵינְךָ: הֲוָיָתִי וַהֲוָיָתְךָ, שֶׁלְּךָ וְשֶׁלִּי,

דַּעְתְּךָ וְדַעְתִּי, אֱמוּנָתְךָ וֶאֱמוּנָתִי,

אֲמִיתְךָ וַאֲמִתִּי, צִדְקָתְךָ

וְצִדְקָתִי.

 

    *  

וְאוּלַי אֶפְשָׁר לְהִתְפַּיֵּס בְּשָׁלוֹם

גַּם עִם הַשְּׁאֵלָה הַצּוֹעֶקֶת...

 

כַּאֲשֶׁר מְקַבְּלִים אֶת הַחַיִּים וּמִתְמַסְּרִים לָהֶם

אֲזַי יְכוֹלִים גַּם לְהִתְעַלּוֹת

מֵעַל צוֹרְכֵי הַ"אֶגוֹ",

וְלֹא לַעֲשׂוֹת עִנְיָן פְּנִימִי כָּל כָּךְ גּוֹרֵף

מִכָּל רֶוַח וּמִכָּל הֶפְסֵד,

מִכָּל הַצְלָחָה וּמִכָּל כִּשָּׁלוֹן...

*

וְאולי אֶפְשָׁר לְהִתְמוֹדֵד קְצָת יוֹתֵר טוֹב

עִם הַשְּׁאֵלָה הַנּוֹרָאִית:

"לָמָּה זֶה קָרָה דַּוְקָא לִי ?"

אוֹ: "לָמָּה דַּוְקָא לִי זֶה לֹא קָרָה ?"

 

וְכַאֲשֶׁר מֻצּוּ כָּל הַבְּרֵרוֹת,

לְקַבֵּל בְּדוּמִיָּה -

ֹ "חֹסֶר מַזָּל" או "מַזָּל רַע".

 

*

 

אַךְ גַּם כַּאֲשֶׁר הַגּוֹרָל מְסַכֵּל לָנוּ

אֶת הַהִזְדַּמְּנֻיּוֹת,

אוֹ שֶׁאֲנַחְנוּ מַחְמִיצִים אוֹתָן;

עֲדַיִן נוֹתֶרֶת לָנוּ הָאֶפְשָׁרוּת

לְהַגִּישׁ

אֶת הַהִזְדַּמְּנֻיּוֹת הַמְּסֻכָּלוֹת וְהַמֻּחְמָצוֹת שֶׁלָּנוּ -

כְּקָרְבָּן לֶאֱלֹהִים.

 

(שמעתי ברדיו את מרסל דיבואה אומר את זה, כשנשאל על חייו האישיים, והאמנתי לו)

 

הַמַּתָּנָה שֶׁאֱלֹהִים מַחְזִיר לָנוּ הִיא

 

קַבָּלַת שְׁלֵמוּת הַחַיִּים

וְקַבָּלַת הַחַיִּים -

בְּשָׁלוֹם.

 

שַׁבַּת שָׁלוֹם

 

אִם הַשַּׁבָּת הִיא אִי שֶׁבֵּין הַזְּמַנִּים -

הַשָּׁלוֹם הוּא אִי שֶׁבֵּין הַלְּבָבוֹת.

אִם הַשַּׁבָּת הִיא שְׁהוּת

בִּמְקוֹם שֶׁהוּא כָּל מָקוֹם -

הַשָּׁלוֹם הוּא שְׁהוּת

בַּיַּחַס שֶׁהוּא כָּל יַחַס.

 

אִם הַשַּׁבָּת הִיא הַקְשָׁבָה לְקוֹל הָעוֹלָם -

הַשָּׁלוֹם הוּא הַקְשָׁבָה לְקוֹל הָאָדָם.

אִם הַשַּׁבָּת הִיא קַבָּלַת מְצִיאוּת הַשַּׁבָּת -

הַשָּׁלוֹם הוּא קַבָּלַת מְצִיאוּת הָאַחֵר.

 

כְּשֵׁם שֶׁהַשַּׁבָּת הָאִידֵאָלִית מַשְׁבִּיתָה

כָּל תְּנַאי וְכָל הַצְדָּקָה

לְחַיִּים שֶׁל עֲשִׂיָּה וְהֶשֵּׂגִים -

כָּךְ הַשָּׁלוֹם הָאִידֵאָלִי מַשְׁבִּית

כָּל תְּנַאי וְכָל הַצְדָּקָה

לַיְּחָסִים הַתְּלוּיִים בַּדָּבָר

וּלְשָׁלוֹם תָּלוּי בַּדָּבָר.

 

הַשָּׁלוֹם הוּא אַהֲבָה, הַשַּׁבָּת הִיא עֹנֶג,

שַׁבַּת-שָׁלוֹם הִנָּהּ אָהֲבָה בַּתַּעֲנוּגִים !

יוסלה שכבר שש פעמים

במשך השבוע הוציאו ממנו דם,

ידיו רזות, אצבעותיו שקופות,

פניו חיוורים כסיד,

וראשו נופל מרוב חולשה על חזהו,

היה לו עוד כח לזכור,

כי מחר בבוקר תהיה זאת

הפעם השביעית

להקזת הדם.

"מחר יום ראשון"

קיווה יוסלה,

"אולי מנגלה ינוח

ממני פעם אחת

ויעשה לעצמו שבת".

אך מנגלה החרוץ,

ממלאכתו לא שבת,

ויוסלה מת

בראשון בשבת.

על אמונה,

מבלי שנוכל לתת איזושהי

משמעות,

למספר הענקי הזה

של אנשים,

שהופחדו, שהורעבו, שנרצחו ושנרטשו

בשואה  ?

 

 

איך אפשר לדבר על אמונה ?

אך אפשר לדבר על תפילה ?

איך אפשר בכלל לדבר,

בשומענו את צעקות המתים  ?

 

         

בכל זאת חיי-לא-שואה ממשיכים,

וילדי לא-שואה נולדים.

כשילדי השואה כבר זקנים.

 

אולי, כמו שהחיים קשורים לזמן,

כמו שהמוות קשור לזמן,

גם החשבון עם השואה,

צריך לקחת בחשבון את הזמן ?

               

לקחת בחשבון את הזמן בהקשר לשואה,

זה אך ורק לדבר על מוות-לא-של-שואה,

כעל מות-לא-של-שואה.

הכמות הרבה והגדולה של אנרגיה שמקבלים מעבודה

כשעושים אותה כנדרש.

ואנרגיה שגודלת ומתגברת כאשר

עושים אותה עם כל הלב.

 

האנרגיה החיובית הזורמת מאיכויות טהורות ואיכויות קדושות.

בניגוד לאנרגיות שליליות הזורמות מאיכויות טמאות ומחוללות.

אנרגיות שקשורות לברכה או לקללה, לעין טובה ועין רעה,

למזל טוב ולמזל רע. 

    

האנרגיה הכי עוצמתית הקשורה לאמונתך,

או להפסד האנרגיה שאתה מפסיד מחוסר אמונתך.

               

כאשר הסיכום והמאזן של נוכחותם או של חסרונם של כל האנרגיות האלה

יכול לתת סימנים אם אלה הם חיים תוססים, חיוניים,

או שהם חיים מנוונים, חיים של חולשה,

עם נטייה למחלה ולמוות.

כאשר תדע את סיבת המוות -

תדע את מידת האחריות של המת למותו,

סיבת המוות כמעט ואיננה מקרית,

היא טבעית, היא שגגתית, היא זדונית,

אך איננה מקרית.

כל סיבה טבעית למוות, או כל סיבה רפואית למוות,

גם אם היא עדיין לא מובנת, אך כל עוד אפשר לחקרה,

אי אפשר לחקור

ואי אפשר להבין.

אך למרות שאי אפשר להבין מוות,

קשה להאמין שהוא מקרי.

כשם שאת זיק החיים,

אי אפשר לחקור

ואי אפשר להבין,

ובכל זאת קשה להאמין

שזיק החיים הוא מקרי.

כַּמָּה שֶׁקִּבֵּל רָ-צָ-ה  ע-וֹ-ד,  עַד שֶׁרְצוֹנוֹ פָּשׁוּט נִשְׁחָק.

 

 *  הוּא אַף פַּעַם  ל-א   י-ד-ע   מ-ה 

ל-ר-צ-ו-ת , עַד שֶׁכְּבָר לֹא הָיָה לוֹ מָה לִרְצוֹת.  (אוּלַי עֵקֶב שִׁטְחִיּוּת, אוֹ עַצְלָנוּת, אוֹ חֹסֶר אִרְגּוּן פְּנִימִי).

                                         

*   הָיוּ לוֹ רְצוֹנוֹת, אַךְ הָרָצוֹן שֶׁלּוֹ הִתְנַוֵּן  מֵאַחַר  שֶׁ-ל-א   י-כ-ל  לְ-הַ-ח-לְ-י-ט  (בִּגְלַל חֻלְשָׁה עֶרְכִּית, אוֹ בִּגְלַל חֲרָדָה קִיצוֹנִית)

 

* רְצוֹנוֹ נִשְׁחַק עַד תֹּם בִּגְלַל  ר צ ו נ ו ת 

מ נ ו ג ד י ם (אוּלַי בְּקוֹנְפְלִיקְט פתלוגי)

 

*  הָרָצוֹן  הַ-ל-ך  וְ-ה-ת-מ-ע-ט , עֵקֶב חֹסֶר תִּקְוָה תְּמִידִי 

(כְּגוֹן בְּמַצָּב שֶׁל יֵאוּשׁ וְדִכָּאוֹן).

 

*   כַּאֲשֶׁר ה-ע-ד-ר  הַ-ר-צ-ו-ן  הוּא בִּטּוּי שֶׁל מַחֲלָה  (כְּמוֹ סְכִיזוֹפְרֶנְיָה).

 

 *  כְּשֶׁהָאָדָם  חַי אֶת חַיָּיו  לְ-ל-א  ר-צ-ו-ן  פ-נ-י-מ-י ,  כְּשֶׁהוּא חַי אֶת חַיָּיו רַק ע"פ הַגֵּרוּי הַ-ח-י-צ-ו-נ-י  הָרִגְעִי .  חַיִּים שֶׁכָּאֵלֶּה הִנָּם חַיִּים שֶׁל חַיָּה לֹא שֶׁל אָדָם, עַל כֵּן הֵם הִגִּיעוּ לְסוֹפָם עוֹד לִפְנֵי תְּחִלָּתָם.    (לְדֻגְמָא, הַפְּסִיכוֹפָּת).

ב.  תְּנָאִים שֶׁל עֲשִׂיָּה   (DOING):

     

הָאָדָם מֵת כִּי:

*  אוֹ שֶׁהוּא כְּבָר עָשָׂה אֶת מָה שֶׁהָיָה

 צ-ר-י-ך,  ר-צ-ה  ו-י-כ-ל לַעֲשׂוֹת.

 

* אוֹ   ש-י-ו-ת-ר  מִמָּה שֶׁעָשָׂה הוּא כְּבָר

  ל-א  י-ע-ש-ה.

 

*  אוֹ  ש-ל-א  ע-ש-ה  ו-ל-א  י-ע-ש-ה.

 

*  אוֹ שֶׁהוּא רַק  כ-א-י-ל-ו   ע-ו-ש-ה   (זוֹרֵעַ רוּחַ וְקוֹצֵר סוֹפָהּ).

 

*  אוֹ שֶׁהוּא  מ-ו-נ-ע  אֲחֵרִים  מִלַּעֲשׂוֹת.

 

*  אוֹ שֶׁמַּעֲשָׂיו  מ-ז-י-ק-י-ם   (מְלַבָּה אֵשׁ וּמַדְלִיק שְׂרֵפוֹת).

 

*  אוֹ שֶׁמַּעֲשָׂיו  ר-ע-י-ם   (גּוֹרֵם רַע וּמֵבִיא חֻרְבָּן).

 

*  אוֹ שֶׁהוּא  ל-א  י-כ-ו-ל  ל-ע-ש-ו-ת  (לְדֻגְמָא עֵקֶב נָכוּת, מַחֲלָה) וְהוּא לֹא מְסֻגָּל  ל-ה-י-ו-ת   בְּלִי  עֲשִׂיָּה  (רְאֵה סְעִיף הָאַחֲרוֹן בַּפֶּרֶק הַבָּא)

 

   ג. תְּנָאִים שֶׁל הֲוָיָה  (BEING):

 

הָאָדָם מֵת כִּי:

* הוּא  כְּבָר  מ-י-צ-ה   א-ת   ה-י-ו-ת-ו. 

 

*  הָאָדָם מִזֵּג אֶת כָּל רְצוֹנוֹ וּמַעֲשָׂיו בִּהְיוֹתוֹ  (בַּהֲוָיָתוֹ).  וְ-הַ-ת-מ-ז-ג  ע-ם  ה-י-ק-ו-ם. (לְדֻגְמָא, חַבְרֵי הָאֶחָד וְהַיָּחִיד מְנַחֵם/מֶנִי רגנסבורגר  זִכְרוֹנוֹ לִבְרָכָה, שֶׁטָּבַע בִּצְעִירוּתוֹ. וְשֶׁעַד טְבִיעָתוֹ מִמֵּשׁ אֶת עַצְמוֹ בְּאִינְטֶנְסִיבִיּוּת מַדְהִימָה. לַמְרוֹת שֶׁשּׁוּם הַשְׁעָרָה לְגַבֵּי מִמּוּשׁוֹ הָעַצְמִי, לְעוֹלָם לֹא יְכוֹלָה לְהַצְדִּיק אוֹ לְהַסְבִּיר אֶת מוֹתוֹ שֶׁל חֲבֵרִי  , שֶׁכָּל כָּךְ הַרְבֵּה הִתְאַבַּלְתִּי עָלָיו וְשֶׁכָּל פַּעַם שֶׁאֲנִי נִזְכָּר בּוֹ עוֹלֶה בִּי מֵחָדָשׁ הָעֶצֶב עַל אָבְדָנוֹ. אַךְ לַמְרוֹת זֹאת נוֹתַרְתִּי עִם הַנֶּחָמָה שֶׁהוּא אָמָּנָם הִגִּיעַ לִשְׁלֵמוּת עַצְמִית, כְּפִי שֶׁהִתְרַשַּׁמְתִּי מִדְּבָרִים שֶׁאָמַר, הִסְבִּיר

וְכָתַב לִי).

 

 *   האדם  ל-א   ה-י-ה   ו-כ-ב-ר   ל-א  

י-ה-י-ה.  דְּהַיְנוּ שֶׁהוּא פִסְפֵס לַחֲלוּטִין אֶת הֲוָיָתוֹ. 

 

*    אֵין לוֹ אֶת   "ה-א-ו-מ-ץ   ל-ה-י-ו-ת".  ("הַהֲוָיָה לְעוֹלָם אֵינֶנָּה נְתוּנָה אוֹטוֹמָטִית בָּאָדָם, אֶלָּא תְּלוּיָה בָּאֹמֶץ שֶׁל הַיָּחִיד וּלְלֹא אֹמֶץ מְאַבְּדִים אֶת הַהֲוָיָה",

נֶאֱמַר ע"י פָּאוּל טִילִיך, מְצֻטָּט ע"י רולו מיי בסיפרו "גִּלּוּי הַהֲוָיָה").

 

*  האדם  ל-א נמצא  ב-מ-צ-ב  ש-ל  ל-ה-י-ו-ת.

עַד עַתָּה, הוּא   ה י ה   רַק בְּמַעֲשָׂיו, אַךְ מֵעַתָּה, נִמְנָע מִמֶּנּוּ יוֹתֵר לַעֲשׂוֹת (עֵקֶב נָכוּת, מַחֲלָה קָשָׁה, זִקְנָה וכד')

וְכָל מָה שֶׁנּוֹתָר לוֹ וְאֶפְשָׁר לוֹ עַכְשָׁו, זֶה רַק  ל ה י ו ת,

אַךְ אוֹ שֶׁהוּא   ל א  ר ו צ ה, בְּמוּדָע אוֹ שֶׁלֹּא בְּמוּדָע,  רַק  

ל ה י ו ת   מִבְּלִי לַעֲשׂוֹת, אוֹ  שֶׁהוּא 

אֵינֶנּוּ  י ו ד ע  אֵיךְ זה  ל ה י ו ת  לְלֹא עֲשִׂיָּה.  (מִפְּנֵי שֶׁכְּאָדָם שֶׁחַיָּיו הָיוּ חַיִּים שֶׁכֻּלָּם מֻקְדָּשִׁים רַק  ל ע ש י י ה  הוּא מֵעוֹלָם לֹא הִתְנַסָּה לִהְיוֹת בַּמַּצָּב שֶׁל  ה ו ו י ה).

                    *

יְּרִידָה בַּדִּמּוּי, בָּעֵרֶךְ וּבַכָּבוֹד הָעַצְמִי

 

הֱיוֹת שֶׁהַגּוּף וְהַנֶּפֶשׁ הִנָּם שְׁנֵי צְדָדִים שֶׁל אוֹתוֹ הַמַּטְבֵּעַ עִם הַשְׁפָּעוֹת הֲדָדִיּוֹת מַכְרִיעוֹת.  הַיְּרִידָה בַּדִּמּוּי, בָּעֵרֶךְ וּבַכָּבוֹד הָעַצְמִי , מְצַמְצֶמֶת אֶת הַהֲוָיָה הַנַּפְשִׁית  וְחוֹנֶקֶת אוֹתָהּ, כַּאֲשֶׁר קָרוֹב לְוַדַּאי, בְּמֶשֶׁךְ הַזְּמַן בְּמַקְבִּיל נֶחְנֶקֶת גַּם הַהֲוָיָה הַפִיזִית.

 

                  *

חֹסֶר תִּקְוָה  

 

   כַּאֲשֶׁר הָאָדָם הוּא בְּכִי רַע, עַל סַף הַשְּׁבִירָה -

שֶׁלֹּא יְרַפֶּה מֵהַתִּקְוָה.

"חֲזַק וְאַמֵּץ לִבְּךָ וְקַוֵּה אֶל הַשֵּׁם"

(תְּהִלִּים כז, יד)

 

אוּלָם כַּאֲשֶׁר חֹסֶר תִּקְוָה מוֹתִירָה את האדם בְּמַצָּב שֶׁל מַשְׁבֵּר כְּרוֹנִי, אֲזַי הוּא נִמְצָא בְּתוֹךְ מִנְהָרָה חֲשׁוּכָה שֶׁל דִּכָּאוֹן 

עִם יְרִידָה קִיצוֹנִית שֶׁל רָצוֹן הַחַיִּים

וְחֹסֶר כּוֹחוֹת לִדְחוֹת

אֶת הַיָּד הַמּוּשֶׁטֶת שֶׁל הַמָּוֶת.

 

          *

הַחַיִּים  הֵם פִּצּוּי מַתְמִיד,

הַחַיִּים נִגְמָרִים -

כַּאֲשֶׁר אֵין בָּהֶם יוֹתֵר תִּקְוָה לְפִּיצוּי.

 

   (בְּתוֹדָה לַפְּסִיכוֹלוֹג הַיָּקָר אלפרד אַדְלֶר - מְכַבֵּד הָאֲנָשִׁים)

 

 עִקָּרוֹן הַפִּצּוּי:  כַּאֲשֶׁר קַיָּם אוֹרְגָּן פִיזִי אוֹ נַפְשִׁי לָקוּי - קַיָּם דָּחַף לְפַצּוֹתוֹ.

חַכְמַת הַנֶּפֶשׁ לְרַפֵּא אֶת עַצְמָהּ מִתְאַפְשֶׁרֶת עַל יְדֵי "עִקָּרוֹן הַפִּצּוּי".

כַּאֲשֶׁר:  "סְטִיָּה  לְלֹא פִּצּוּי  גּוֹרֶמֶת לְתֹהוּ וַבוֹהוּ נַפְשִׁי וְהֶמְשֵׁכָהּ בְּאֶפְשָׁרוּת שֶׁל זֵרוּז תְּמוּתַת הַגּוּף" !  (הַמִּשְׁפָּט מְצֻטָּט מֵהַסֵּפֶר: "חָכְמַת הַנֶּפֶשׁ לְרַפֵּא אֶת עַצְמָהּ", מֵאֵת י. וַרְדִי).  

                     *

כְּשֶׁהַרְגָּשַׁת הַחַיִּים נְחוּתָה -

מְפַצָּה הַשְּׁאִיפָה לְהַרְגָּשָׁה עֶלְיוֹנָה,

כְּשֶׁהַמְּצִיאוּת קָשָׁה - הַתִּקְווֹת וְהַחֲלוֹמוֹת מְפַצִּים.

כְּשֶׁמַּרְגִּישִׁים חֹסֶר מַשְׁמָעוּת בַּחַיִּים - אֱמוּנָה, עֲבוֹדָה, מַדָּע וִיצִירָה מְפַצִּים.

כַּאֲשֶׁר אֵין עֲבוֹדָה וְאֵין יְצִירָה -  הַיְּדִידוּת מְחַזֶּקֶת.

כַּאֲשֶׁר הָאָדָם בּוֹדֵד לְנַפְשׁוֹ - הַקֶּשֶׁר שֶׁלּוֹ עִם אֱלֹהָיו מְפַצֶּה.

אַךְ כַּאֲשֶׁר אֲפִלּוּ אֱלֹהִים מַכֶּה (לְעוֹלָם אַל תִּשְׁכַּח אֶת אושוויץ !) -

רַק הַמָּוֶת מְפַצֶּה.

כְּשֶׁהַמָּוֶת מְכַסֶּה  - (הַלְוַאי שֶׁ-)  הַחַיִּים שֶׁלְּאַחֲרָיו יְפַצּוּ.

 

וְגַם כְּשֶׁהַהַרְגָּשָׁה נְחוּתָה,

וְגַם כְּשֶׁהַמְּצִיאוּת רָעָה,

וְגַם כְּשֶׁהַחֲלוֹמוֹת רָעִים,

וְגַם כַּאֲשֶׁר אֵין עֲבוֹדָה וְאֵין יְצִירָה,

וְגַם כַּאֲשֶׁר הָאָדָם בּוֹדֵד,

וַאֲפִלּוּ שֶׁאֱלֹהִים מַכֶּה-

כָּל עוֹד, כָּל עוֹד הַמָּוֶת לֹא מְכַסֶּה -

תַּמְשִׁיךְ לְקַוּוֹת.

מָה שֶׁלֹּא יִהְיֶה, אַף פַּעַם אַל תֶּחְדַּל מִלְּקַוּוֹת,

קַוֵּה תָּמִיד! תָּמִיד תְּקַוֶּה !

כָּל הַזְּמַן תְּעוֹרֵר אֶת הַתִּקְוָה וְכָל הַזְּמַן תַּעֲצִים וְתַמְשִׁיךְ עוֹד וָעוֹד לְהַעֲצִים - - -

כִּי כָּל עוֹד הָאָדָם חַי יֵשׁ לוֹ תִּקְוָה.

וְסוֹף הַתִּקְוָה לְפַצּוֹת עַל הַכֹּל,

הַתִּקְוָה הִיא אֵם-כָּל-פִּצּוּי !

              *

גְּבוּלוֹת הָאַחְרָיוּת:

 

כַּאֲשֶׁר תֵּדַע אֶת סִבַּת הַמָּוֶת -

תֵּדַע אֶת מִדַּת הָאַחְרָיוּת שֶׁל הַמֵּת לְמוֹתוֹ,

אוֹ שֶׁל זוּלָתוֹ שֶׁגָּרַם לְמוֹתוֹ שֶׁל הַמֵּת.

 

סִבַּת הַמָּוֶת כִּמְעַט וְאֵינֶנָּה מִקְרִית,

הִיא טִבְעִית, הִיא שגגתית, הִיא זְדוֹנִית, אַךְ אֵינֶנָּה מִקְרִית.

כָּל סִבָּה טִבְעִית לַמָּוֶת, אוֹ כָּל סִבָּה רְפוּאִית לַמָּוֶת, גַּם אִם הִיא עֲדַיִן לֹא מוּבֶנֶת,

אַךְ כָּל עוֹד אֶפְשָׁר לְחָקְרָהּ,

סָבִיר לְהָנִיחַ שֶׁאֶפְשָׁר יִהְיֶה גַּם לְהָבִינָהּ.

לָכֵן, הִיא אֵינֶנָּה מִקְרִית.

רַק אֶת הַמָּוֶת עַצְמוֹ -

אִי אֶפְשָׁר לַחְקֹר

וְאִי אֶפְשָׁר לְהָבִין.

אַךְ לַמְרוֹת שֶׁאִי אֶפְשָׁר לְהָבִין מָוֶת,

קָשֶׁה לְהַאֲמִין שֶׁהוּא מִקְרִי.

כְּשֵׁם שֶׁאֶת זִיק הַחַיִּים,

אִי אֶפְשָׁר לַחְקֹר

וְאִי אֶפְשָׁר לְהָבִין,

וּבְכָל זֹאת קָשֶׁה לְהַאֲמִין

שֶׁזִּיק הַחַיִּים הוּא מִקְרִי.

פרק י' : הַגוֹלֶם

 

הוּא מורכב מֵתאים.

הַרְבֵּה תאים, הַרְבֵּה מְאֹד תאים.

לֵךְ לְתוֹכְנִית מס' 1 בְּסֵפֶר הַגֹּלֶם,

וּרְאֵה אֶת מִסְפַּר הָתאים,

בְּדֹק לִפְנֵי זֶה שֶׁאַינך חוֹלֵם...

עשרות טריליונים, לא פחות.

הוֹמֵאוֹסְטָזִיס או מַצַּב אִזּוּן או מצב יציב (מיוונית: Homeostasis) הוא היכולת של האורגניזם לשמור על סביבה פנימית יציבה למרות השינויים שחלים בסביבה החיצונית. זהו שם גג למכלול של תהליכים ביוכימיים-מטבוליים, המשמרים רמה יציבה של חומרים המומסים בנוזלי הגוף, גם כאשר שינויים חלים בסביבה החיצונית ובסביבה הפנימית.

תאי הגוף מאורגנים כרקמות שונות: רקמת עצב, רקמת שריר, רקמת חיבור ורקמת אפיתל, המרכיבות את איברי הגוף. נהוג לקבץ איברים שונים למערכות גוף הממלאות תפקידים שונים. תפקידן של מערכות הגוף הוא שימור ההומאוסטזיס בסביבת התאים (בנוזל החוץ-תאי) והן עושות זאת בין היתר על ידי הכנסת חומרי מזון לנוזל הבין-תאי, שינוע הנוזל לכל תאי הגוף, סילוק פסולת או תוצרים סופיים של המטבוליזם התאי ושליטה ובקרה על פעולות אלו ואחרות.

 

הָאָטוֹמִים הַמִּתְחַבְּרִים בְּצוּרוֹת שׁוֹנוֹת,

יוֹצְרִים בְּהִתְחַבְּרוּתָם, מוֹלֶקוּלוֹת שׁוֹנוֹת

שֶׁל חֳמָרִים שׁוֹנִים: בָּרֹאשׁ וּבָרִאשׁוֹנָה תָּאִים,

אַחֲרֵיהֶם, חֻמְצוֹת, אֶנְזִימִים וְהוֹרְמוֹנִים.   

 

לֵךְ לְתוֹכְנִית  מס' 6 בְּסֵפֶר הַגֹּלֶם,

וּרְאֵה אֶת הַתוֹכְנִית הַגֶּנֶטִית הַסְּפֶּצִיפִית.

רְאֵה אֵיךְ נִבְנֵית רִקְמַת הָעֶצֶם,

רִקְמַת הַשְּׁרִיר, וְרִקְמַת הַחִבּוּר הַהֶקֵּפִית.

 

אֵיךְ מַתְוִים אֶת תָּוֵי הַפָּנִים וְהַמַּבָּע הַחַד פַּעֲמִי,

הַגֹּבַהּ, גֵּוָן הַשֵּׂעָר וְצֶבַע הָעֵינַיִם.

הַשֵּׂכֶל הַטּוֹב, הָרֶגֶשׁ הַחַם וְהַכִּשָּׁרוֹן הַחַד פַּעֲמִי,

וַאֲפִלּוּ אֶת תְּוַי אֶצְבְּעוֹת הַיָּדַיִם.

 

וּבַאֲשֶׁר לְחֻקֵּי הַהַנְדָּסָה הַפְּסִיכוֹלוֹגִית:

חֹק רִאשׁוֹן: תִּסְכּוּל מְעוֹרֵר אַגְרֶסְיָה.

לְפִי חֹק זֶה אֶפְשָׁר לְמָשָׁל לְהַבְהִיר:  

מַדּוּעַ כַּאֲשֶׁר צַעֲקָתוֹ שֶׁל הָאָדָם

לְחֶסֶד וְתִקְוָתוֹ לִישׁוּעָה -

נִשְׁאֶרֶת בָּאֲוִיר, הוּא יוֹצֵא מֵאִפּוּס...

חֹק שֵׁנִי: חֹק הסוציאליזיה,

לָדַעַת לְעַכֵּב צְרָכִים וְלִדְחוֹת סִפּוּקִים.

עַל פִּי חֹק זֶה לְמָשָׁל, אֶפְשָׁר לְהָבִין בָּאֱמֶת,

מַדּוּעַ מִתְכּוֹנֵן הַסְּטוּדֶנְט לִבְחִינַת הַגְּמָר שֶׁלּוֹ

עִם רוֹבֶה טָעוּן לְצִדּוֹ:

אוֹ שֶׁהוּא עוֹבֵר אוֹ שֶׁהוּא מֵת...

וּבַל נִשְׁכַּח אֶת חֹק: הַמַּקֵּל וְהַגֶּזֶר.

כָּךְ בְּעֵרֶךְ מִשְׁתַּלֶּטֶת הַמּוֹרָה שֶׁל כִּתָּה אָלֶף.

בְּדֶרֶךְ כְּלָל זֶה עוֹזֵר, 

כִּי אַחֲרֵי הַחִנּוּךְ שֶׁלָּהּ, כִּמְעַט כָּל אֶחָד יוֹדֵעַ:

שֶׁעָדִיף לִשְׁתֹּק מִלְּדַבֵּר...

בַּמִּשְׁפָּחָה הַהוֹרִים מַפְעִילִים הוֹרוּת

כְּמוֹ שֶׁמַּפְעִילִים מַשְׁאֵבָה עַל בְּאֵר,

בְּהַשְׁפִּיעָם אוֹ בְּמוֹנְעָם אַהֲבָה וְהוֹרוּת.

לְעֻמָּתָם, אֱלִיל הַנֹּעַר יוֹתֵר מְתֻחְכָּם,

הוּא מַפְעִיל אֶת חֹק הַהִזְדַּהוּת.

    

קָרָאנוּ בְּסֵפֶר הַגֹּלֶם, אֶת רָאשֵׁי הַפְּרָקִים

שֶׁל כָּל הַתָּכְנִיּוֹת הַמַּרְכִּיבוֹת אוֹתוֹ:

הַתָּכְנִית הַפִיסִיקָלִית, הַתָּכְנִית הַכִימִית,

הַתָּכְנִית הַבִּיוֹכִימִית, הַתָּכְנִית הַגֶּנֶטִית,

הַתָּכְנִית הַבִּיוֹלוֹגִית, הַתָּכְנִית הָאָנָטוֹמִית,

הַתָּכְנִית הַפִיזְיוֹלוֹגִית, הַתָּכְנִית הַקּוֹגְנִיטִיבִית

וְהַתָּכְנִית הַפְּסִיכוֹלוֹגִית.

 

כַּאֲשֶׁר כָּל הַתוֹכְנִיּוֹת הָאֵלֶּה מוּצָאוֹת אֶל הַפֹּעַל

בְּדַיְקָנוּת מְרַבִּית -

קָם הַגֹּלֶם.

.

אַךְ גֹּלֶם נִשְׁאַר גֹּלֶם!

תָּמִיד בַּמַּעְגָּל שֶׁל עַצְמוֹ.

 

צֵא צֵא מִן הַמַּעְגָּל צָרִיךְ לָשִׁיר לוֹ,

גֹּלֶם צֵא מֵהַמַּעְגָּל...        

 

 

פרק יב. מֵעֵבֶר לַזְּמַן וּמֵעֵבֶר לַגּוּף  

 

מֵעֵבֶר לַזְּמַן וּמֵעֵבֶר לַגּוּף

  

לַמּוֹחַ, זְמַן מָוֶת אֶחָד,

לַלֵּב,  זְמַן מָוֶת שֵׁנִי,

לַבָּשָׂר, זְמַן מָוֶת שְׁלִישִׁי -

הַחַיִּים הֵם חַיִּים בַּזְּמַן.

 

בְּתֹם זְמַנָּם,

הַמֵּת נֶאֱסַף

אֶל עַמָּיו,

זִכְרוֹנוֹתָיו נֶאֱגָרִים בְּצֶאֱצָאָיו.

 

"חַיִּים" שֶׁהֵם מֵעֵבֶר לַזְּמַן,

אוּלַי הֵם נְשָׁמוֹת? אוּלַי כּוֹחוֹת? אוּלַי רוּחוֹת?

אֵיךְ שֶׁלֹּא יִקָּרְאוּ וּמָה שֶׁלֹּא יִהְיוּ,

הֵן מְשַׁנּוֹת אֶת "מַצַּב הַצְּבִירָה"

וּמִי יְדָעָן?

 

 

 אַרְבָּעָה מְשָׁלִים עַל חֹמֶר וְרוּחַ

 

  1.   הַמֶּלֶךְ וְהַמַּמְלָכָה

 

הַמֶּלֶךְ מֵת, הַמְּשָׁרְתִים אַחֲרָיו,

הָאַרְמוֹן נֶהֱרַס -

אֵפוֹא הַמַּמְלָכָה ?

 

 2.   הַמַּחְשֵׁב וְהַתוֹכְנָה    

                

הַמַּחְשֵׁב מִתְקַלְקֵל, חַשְׁמַל לֹא עוֹבֵר בַּחוּטִים, הַפַּח מִתְקַמֵּט -

מָה עִם הַתוֹכְנָה?

 

3.   הַכַּנָּר וְהַיְּצִירָה

 

הַכַּנָּר מִשְׁתַּתֵּק, מֵיתְרֵי הַכִּנּוֹר פּוֹקְעִים,

הַקֶּשֶׁת נִשְׁבֶּרֶת,

אָבְדוּ הַתָּוִים -

וְהַיְּצִירָה מַמְשִׁיכָה לְהַדְהֵד...

                

   

 4.   הָחוֹמָרִים וְהַנּוֹף

          

מִישֶׁהוּ הוֹלֵךְ בְּמִקְרֶה אוֹ בְּכַוָּנָה,

לְמָקוֹם מִקְרִי אוֹ לְמָקוֹם מְכֻוָּן,

מָקוֹם בּוֹ, נִמְצָאִים:

אֲבָנִים בִּגְדָלִים שׁוֹנִים וּבְצוּרוֹת שׁוֹנוֹת,

מִפְלְסֵי אֲדָמָה בִּגְבָהִים מְסֻיָּמִים,

עָפָר בִּתְחִיחוּת מִשֶּׁלּוֹ: חוֹל, חֵמָר, אוֹ לֶס

וְכַמָּה זַנִּים וּמִינִים שֶׁל צְמָחִים.

 

כַּאֲשֶׁר אוֹתוֹ הָאָדָם,

בְּאוֹתוֹ הַמָּקוֹם,

רוֹצֶה לִרְאוֹת,

לְהִתְרַשֵּׁם

וּלְהֵנוֹת

מֵהַנּוֹף;

 

מִיָּד -

הָאֲבָנִים פּוֹסְקִים מִלִּהְיוֹת

רַק אֲבָנִים,

הֶעָפָר פּוֹסֵק מִלִּהְיוֹת

רַק עָפָר,

הַצְּמָחִים פּוֹסְקִים מִלִּהְיוֹת

רַק צְמָחִים.

וְכָל הַמַּרְכִּיבִים הַלָּלוּ

מִתְאַרְגְּנִים כְּהֶרֶף עַיִן וְהוֹפְכִים

לְנוֹף מַרְהִיב.

 

אַךְ הַנּוֹף, הַנּוֹף הַזֶּה הַמַּרְהִיב, הוּא כְּשֶׁלְּעַצְמוֹ,

אֵיפֹה הוּא? מִנַּיִן הוּא בָּא

וּלְאָן הוּא הוֹלֵךְ? 

 

הֲגַם כַּאֲשֶׁר יֵצֵא הָאִישׁ 

מֵאוֹתוֹ הַמָּקוֹם,

יִשָּׁאֵר הַנּוֹף בְּאוֹתוֹ הַמָּקוֹם -

אוֹ שֶׁמָּא יִקַּח הָאִישׁ אֶת הַנּוֹף עִמּוֹ

בְּדִמְיוֹנוֹ, בְּזִכְרוֹנוֹ -

וְהַמָּקוֹם יַחְזֹר לִהְיוֹת הַמָּקוֹם

שֶׁל הֶעָפָר הַהוּא וְשֶׁל

אוֹתָם הָאֲבָנִים

וְהַצְּמָחִים?

 

     

יג. הַמַּהוּת הַנִּשְׁאֶרֶת

 

כַּאֲשֶׁר הַחַי מֵת,

אוּלַי -

קַיֶּמֶת אַחֲרָיו מַהוּת

שֶׁנִּשְׁאֶרֶת,

כַּאֲשֶׁר הַחַי מֵת,

הַלְוַאי -

שֶׁקַּיֶּמֶת אַחֲרָיו מַהוּת

שֶׁנִּשְׁאֶרֶת !

תִּהְיֶה מָה שֶׁתִּהְיֶה -

הַמַּהוּת הַנִּשְׁאֶרֶת,

הִיא אַחֶרֶת מִמַּהוּת הַחַיִּים !

אַתָּה בַּמֶּדִיטַצְיָה

אֲנַחְנוּ לֹא עוֹצְרִים.

 

אַתָּה זוֹרֵם בְּשַׁלְוָה מִנְּקוּדָה לִנְקוּדָה -

אֲנַחְנוּ קוֹפְצִים בְּעַצְבָּנוּת מִנְּקוּדָה לִנְקוּדָה.

אַתָּה מוּדָע לַנְּקוּדָה וְחוֹוֶה אוֹתָהּ -

אֲנַחְנוּ מְצַיְּנִים אוֹתָהּ כְּעוֹד נ.צ.

 

מַקִּיפִים אוֹתָהּ בְּגָדֵר

וְשָׂמִים שֶׁלֶט.

מַצִּיבִים עָלֶיהָ תֹּרֶן,

וּמְנִיפִים ת'דֶּגֶל.

 

אֲנַחְנוּ חַיִּים עִם נֻסְחֲאוֹת-

אַתָּה חַי סְפּוֹנְטָנִית,

אֲנַחְנוּ נִגָּשִׁים עִם רְשִׁימוֹת-

אַתָּה מֵגִיב טִבְעִית.

 

אַתָּה צוֹפֶה-

אֲנַחְנוּ מְעֹרָבִים,

אַתָּה צוֹפֶה-

אֲנַחְנוּ מִזְדַּהִים.

 

אֶצְלְךָ, הַמַּעֲשִׂים הַנַּעֲשִׂים-

הֵם תִּפְקוּדִיִּים וּתְלוּיֵי מַצָּב,

אֶצְלֵנוּ, הַמַּעֲשִׂים הַנַּעֲשִׂים-

הֶשֵּׂגִיִּים, מַטְרָתִיִּים, אוֹ תַּחַת צַו.

                        

אֶצְלְךָ, מַרְאוֹת נִרְאִים,

מַחְשָׁבוֹת נֶחְשָׁבוֹת, מַעֲשִׂים נַעֲשִׂים...

אֶצְלֵנוּ, אֲנַחְנוּ רוֹאִים!

אֲנַחְנוּ חוֹשְׁבִים! אֲנַחְנוּ עוֹשִׂים!

 

בִּתְחוּשָׁתְךָ-

קַיֶּמֶת רַק מוּדָעוּת,

בִּתְחוּשָׁתֵנוּ-

רַק אֲנַחְנוּ קַיָּמִים.

 

בְּעוֹלָמְךָ אֵין דְּמוּת וְאֵין רֶקַע -

הַכֹּל מוּדָעוּת.

בְּעוֹלָמֵנוּ, כָּל מָה שֶׁמִּחוּץ לָנוּ, בָּרֶקַע-

וַאֲנַחְנוּ הַדְּמוּת.

        

אַתָּה שְׁקוּעַ נֶצַח

בַּהֲוָיָה מַחְזוֹרִית

וַאֲנַחְנוּ שְׁלוּחִים כְּמוֹ חֵץ

בַּהֲוָיָה זְמַנִּית.

 

כְּכָל שֶׁהָאֶנֶרְגְּיָה שֶׁלְּךָ מִתְרוֹקֶנֶת

כָּךְ הִיא מִתְמַלֵּאת,

וְאִלּוּ הָאֶנֶרְגְּיָה שֶׁלָּנוּ נִסְחֶטֶת

פַּעַם-אַחַר-פַּעַם לְלֹא שְׂאֵת.

 

בְּשָׁעָה שֶׁאָנוּ תְּקוּעִים בִּנְקֻדַּת

אֵל-חָזוֹר,

אַתָּה מַמְשִׁיךְ לִזְרֹם

בְּמַחְזוֹר.           

 

אֲנַחְנוּ מְבֹהָלִים מֵהַמָּוֶת -   

בִּשְׁבִילְךָ מָוֶת הוּא חֵלֶק מֵהַחַיִּים.

אֲנַחְנוּ מַדְחִיקִים אֶת בּוֹאוֹ

אַתָּה חַי אוֹתוֹ בְּכָל הַחַיִּים.

 

עַל כֵּן בִּשְׁבִילְךָ הַמָּוֶת מִתְחַזֶּה,

וּבִשְׁבִילֵנוּ הוּא אֲמִתִּי.

עַל כֵּן, הִנְּךָ חָשׁ נִצְחִיּוּת

בְּתוֹךְ הַחַיִּים

וְאָנוּ חָשִׁים אֲרָעִיּוּת

בְּתוֹךְ "הָאֲנִי".

 

בְּסוֹף זֶה אֲנַחְנוּ

עִם פַּרְצוּף רְצִינִי שֶׁכָּזֶה

וְאַתָּה מְחַיֵּךְ...

אָז מִי הוּא הָאֲמִתִּי

וּמִי הוּא הַמִּתְחַזֶּה ?

 

 

 

 

י"ז.  סינתזה  בין המודעות לאמונה

כַּאֲשֶׁר הַמּוּדָעוּת הִיא,

תְּשׂוּמַת-לֵב

הַמְּקַבֶּלֶת בִּבְרָכָה

אֶת הַדְּבָרִים

כְּמוֹ שֶׁהֵם.

 

 

אֲזַי הָאֱמוּנָה הִיא,

אֲמִירַת אָמֵן בְּכָל הַכּוֹחַ -

לְמַהוּתָם שֶׁל

הַדְּבָרִים

מוּדָעוּת מְלֵאָה -

הִיא עוֹלָם מָלֵא שֶׁל קֶשֶׁב.

אֱמוּנָה מְלֵאָה -

הִיא עוֹלָם מָלֵא שֶׁל דִּבּוּר.

  

 בְּהַקְשָׁבָה לְדִבּוּר     

מוּדָעוּת מְבִיאָה לֶאֱמוּנָה.

אִם יֵשׁ נְשָׁמָה:

טוֹב מְאֹד הוּא הַמָּוֶת,

וְהַחַיִּים - הֲזָיָה.

 

אִם אֵין נְשָׁמָה:

טוֹבִים מְאֹד הֵם הַחַיִּים,

וְהַמָּוֶת - עֻבְדָּה.

 

אִם יֵשׁ נְשָׁמָה,

וְאִם אֵין נְשָׁמָה -

טוֹבִים, טוֹבִים מְאֹד הֵם הַחַיִּים !
